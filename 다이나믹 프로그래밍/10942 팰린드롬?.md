def is_palindrome(s, e):
    global arr, dp
    if dp[s][e] != -1:
        return dp[s][e]
    if s >= e:
        dp[s][e] = 1
        return 1
    if arr[s] != arr[e]:
        dp[s][e] = 0
        return 0
    dp[s][e] = is_palindrome(s + 1, e - 1)
    return dp[s][e]

            return 1
    if dp[s][e] >= 0:
        return dp[s][e]
    if a[s] != a[e]:
        dp[s][e] = 0
        return dp[s][e]
    dp[s][e] = palindrome(s + 1, e - 1)
    return dp[s][e]
