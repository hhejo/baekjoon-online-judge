import sys

input = sys.stdin.readline

def is_palindrome(s, e):
    global arr, dp
    if dp[s][e] != -1:
        return dp[s][e]
    if s >= e:
        dp[s][e] = 1
        return 1
    if arr[s] != arr[e]:
        dp[s][e] = 0
        return 0
    dp[s][e] = is_palindrome(s + 1, e - 1)
    return dp[s][e]


N = int(input())
arr = [0] + list(map(int, input().split()))
M = int(input())
dp = [[-1] * 2001 for _ in range(2001)]
ans = [-1] * M
for i in range(M):
    S, E = map(int, input().split())
    ans[i] = is_palindrome(S, E)
print(*ans, sep='\n')
