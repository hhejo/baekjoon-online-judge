import sys

input = sys.stdin.readline


# 풀이
def solve():
    global N
    dp = [0] * 101
    dp[1] = 1
    dp[2] = 1
    dp[3] = 1
    dp[4] = 2
    dp[5] = 2
    dp[6] = 3

    for i in range(7, N + 1):
        dp[i] = dp[i - 1] + dp[i - 5]

    ans = dp[N]
    print(ans)
    return


for test_case in range(int(input())):
    N = int(input())
    solve()
