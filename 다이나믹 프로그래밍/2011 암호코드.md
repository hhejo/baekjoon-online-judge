## CODE

```python
import sys

input = sys.stdin.readline

S = input().strip()
S = ' ' + S


# 풀이
def solve():
    global S
    MOD = 1_000_000
    dp = [0] * 5001
    res = 0
    if S[1] == '0':
        print(res)
    else:
        dp[0] = dp[1] = 1
        for i in range(2, len(S)):
            if 1 <= int(S[i]) <= 9:
                dp[i] += dp[i - 1]
            tmp = int(S[i - 1]) * 10 + int(S[i])
            if 10 <= tmp <= 26:
                dp[i] += dp[i - 2]
        res = dp[len(S) - 1]
        print(res % MOD)
    return


solve()

```
