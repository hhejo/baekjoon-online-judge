## CODE 1

```python
import sys

input = sys.stdin.readline

# 입력
N = int(input())
P = [0] + list(map(int, input().split()))


# 풀이
def solve():
    global N, P
    dp = [0] * 1001  # dp[n] : 카드 n개 갖기 위한 최대 금액

    # dp[n] = max(dp[1] + P[n - 1], dp[2] + P[n - 2], ... dp[n] + P[0])
    for i in range(1, N + 1):
        for j in range(1, i + 1):
            # dp[i] 최댓값 갱신
            if dp[i] < dp[i - j] + P[j]:
                dp[i] = dp[i - j] + P[j]

    print(dp[N])  # 정답 출력
    return


solve()

```

## 해설 1

`31256KB`, `76ms`

## CODE 2

```python
import sys

input = sys.stdin.readline

# 입력
N = int(input())
P = [0] + list(map(int, input().split()))


# 풀이
def solve():
    global N, P
    dp = [0] * 1001  # dp[n] : 카드 n개 갖기 위한 최대 금액

    # dp[n] = max(dp[1] + P[n - 1], dp[2] + P[n - 2], ... dp[n] + P[0])
    for i in range(1, N + 1):
        for j in range(1, i + 1):
            dp[i] = max(dp[i], dp[i - j] + P[j])  # dp[i] 최댓값 갱신

    print(dp[N])  # 정답 출력
    return


solve()

```

## 해설 2

`max()` 함수를 이용했는데 시간이 2배 더 걸리게 나왔다.

`31256KB`, `148ms`
