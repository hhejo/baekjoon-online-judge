    global N

    dp = [0] * 11

    def calc(n):
        if dp[n] > 0:
            return dp[n]
        if n == 0 or n == 1:
            return 1
        if n == 2:
            return 2
        dp[n] = calc(n - 3) + calc(n - 2) + calc(n - 1)
        return dp[n]

    ans = calc(N)
    print(ans)
    return
