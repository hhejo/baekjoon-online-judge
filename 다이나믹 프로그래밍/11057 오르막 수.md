## CODE

```python
import sys

input = sys.stdin.readline

# 입력
N = int(input())  # 1 <= N <= 1000 자연수


# 풀이
def solve():
    global N
    MOD = 10_007  # 제수
    dp = [[0] * 10 for _ in range(1001)]  # dp[n][k] : n자리, k로 끝나는 계단수
    # 1자리 계단수 초기화
    for x in range(10):
        dp[1][x] = 1  # 1자리 x로 끝나는 계단수는 전부 하나

    # 2자리 ~ N자리
    for i in range(2, N + 1):
        # 0 ~ 9
        for j in range(0, 10):
            dp[i][j] = sum(dp[i - 1][j:]) % MOD  # i-1자리 j, ... 9로 끝나는 계단수 개수 합

    ans = sum(dp[N]) % MOD
    print(ans)
    return


solve()

```
