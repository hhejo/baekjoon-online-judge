import sys

N = int(input())

dp = [''] * 1001
dp[1] = dp[3] = 'SK'
dp[2] = 'CY'

for i in range(4, N + 1):
    if dp[i - 1] == 'SK' or dp[i - 3] == 'SK':
        dp[i] = 'CY'
    else:
        dp[i] = 'SK'

ans = dp[N]
print(ans)
