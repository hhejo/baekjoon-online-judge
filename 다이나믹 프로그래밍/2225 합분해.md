## CODE 1

```python
import sys

input = sys.stdin.readline

N, K = map(int, input().split())


# 풀이
def solve():
    global N, K
    MOD = 1_000_000_000
    dp = [[0] * 201 for _ in range(201)]
    dp[0][0] = 1

    for k in range(1, K + 1):
        for n in range(N + 1):
            for l in range(n + 1):
                dp[k][n] += dp[k - 1][n - l]
                dp[k][n] %= MOD

    ans = dp[K][N]
    print(ans)
    return


solve()

```

## 해설 1

`32276KB`, `668ms`

## CODE 2

```python
import sys

input = sys.stdin.readline

N, K = map(int, input().split())


# 풀이
def solve():
    global N, K
    MOD = 1_000_000_000
    dp = [[0] * 201 for _ in range(201)]
    dp[0][0] = 1

    for k in range(1, K + 1):
        for n in range(N + 1):
            dp[k][n] = (dp[k - 1][n] + dp[k][n - 1]) % MOD

    ans = dp[K][N]
    print(ans)
    return


solve()

```

## 해설 2

`32276KB`, `48ms`
