## CODE

```python
import sys

input = sys.stdin.readline

N = int(input())
g = list(map(int, input().split()))
D = int(input())


# 풀이
def solve():
    global N, g, D

    # 현재 노드와 연결된 노드 모두 삭제 (-2로 값 변경)
    def delete_nodes(cur_node: int):
        g[cur_node] = -
        # i : 노드 번호, g[i] : 해당 노드의 부모 번호
        for i in range(N):
            if g[i] == cur_node:
                delete_nodes(i)
        return

    # 리프 노드 개수 구하기
    def get_leaves_count():
        cnt = 0
        for i in range(N):
            if i not in g and g[i] != -2:
                cnt += 1
        return cnt

    delete_nodes(D)
    ans = get_leaves_count()
    print(ans)
    return


solve()

```
