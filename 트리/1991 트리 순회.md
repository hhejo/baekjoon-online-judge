# 1991 트리 순회

## CODE

```python
import sys

input = sys.stdin.readline


def to_num(ch):
    return ord(ch) - ord('A')


def to_chr(num):
    return chr(num + ord('A'))


N = int(input())
T = [0] * 26
for i in range(N):
    P, C1, C2 = input().split()
    T[to_num(P)] = (to_num(C1), to_num(C2))


# 풀이
def solve():
    global N, T

    def preorder(node):
        if node < 0:
            return
        print(to_chr(node), end='')
        preorder(T[node][0])
        preorder(T[node][1])
        return

    def inorder(node):
        if node < 0:
            return
        inorder(T[node][0])
        print(to_chr(node), end='')
        inorder(T[node][1])
        return

    def postorder(node):
        if node < 0:
            return
        postorder(T[node][0])
        postorder(T[node][1])
        print(to_chr(node), end='')
        return

    preorder(0)
    print()
    inorder(0)
    print()
    postorder(0)
    print()
    return


solve()

```
