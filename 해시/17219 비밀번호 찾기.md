## CODE

import sys

input = sys.stdin.readline

N, M = map(int, input().split())

passwordDict = {}
for _ in range(N):
    A, B = input().rstrip().split()
    passwordDict[A] = B

res = [''] * M
idx = 0
for _ in range(M):
    A = input().rstrip()
    res[idx] = passwordDict[A]
    idx += 1

print(*res, sep='\n')
