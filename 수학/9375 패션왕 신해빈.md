## CODE

```python
import sys

input = sys.stdin.readline

for _ in range(int(input())):
    N = int(input())
    wearable = {}
    for _ in range(N):
        a, b = input().rstrip().split()
        if wearable.get(b):
            wearable[b].append(a)
        else:
            wearable[b] = [a]
    res = 1
    for k, v in wearable.items():
        res *= len(v) + 1
    ans = res - 1
    print(ans)

```
