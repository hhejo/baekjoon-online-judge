## CODE

```python
import sys

input = sys.stdin.readline

N = int(input())
a = list(map(int, input().split()))


def solve():
    global N, a

    def next_permutaion():
        i = N - 1
        while i > 0 and a[i - 1] >= a[i]:
            i -= 1

        if i <= 0:
            return False

        j = N - 1
        while a[j] <= a[i - 1]:
            j -= 1

        a[i - 1], a[j] = a[j], a[i - 1]

        j = N - 1
        while i < j:
            a[i], a[j] = a[j], a[i]
            i += 1
            j -= 1
        return True

    if next_permutaion():
        print(*a)
    else:
        print(-1)
    return


solve()
```
