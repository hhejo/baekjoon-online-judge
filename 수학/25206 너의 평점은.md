## CODE 1

```python
import sys

input = sys.stdin.readline

a = [list(input().split()) for _ in range(20)]

grade = {'A+': 4.5, 'A0': 4.0, 'B+': 3.5, 'B0': 3.0,
         'C+': 2.5, 'C0': 2.0, 'D+': 1.5, 'D0': 1.0, 'F': 0.0}


acc = res = 0
for x, y, z in a:
    if z == 'P':
        continue
    y = float(y)
    acc += y
    res += y * grade[z]

ans = res / acc
print(f'{ans:.6f}')
```
