## 1.

```python
import sys

while True:
    sentence = sys.stdin.readline().rstrip()
    if sentence == '.':
        break
    stk = []
    ans = 'yes'
    for ch in sentence:
        if ch in ['(', '[']:
            stk.append(ch)
        elif ch == ')':
            if stk:
                if stk[-1] == '(':
                    stk.pop()
                else:
                    ans = 'no'
                    break
            else:
                ans = 'no'
                break
        elif ch == ']':
            if stk:
                if stk[-1] == '[':
                    stk.pop()
                else:
                    ans = 'no'
                    break
            else:
                ans = 'no'
                break
    if stk:
        ans = 'no'
    print(ans)

```

import sys

            if stk and stk[-1] == '(':
                stk.pop()
            else:
                ans = 'no'
                break

            if stk and stk[-1] == '[':

            else:
                ans = 'no'
                break
