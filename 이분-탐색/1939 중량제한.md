```python
import sys
from collections import deque

input = sys.stdin.readline

N, M = map(int, input().split())
g = [[] * (N + 1) for _ in range(N + 1)]
for _ in range(M):
    A, B, C = map(int, input().split())
    g[A].append((B, C))
    g[B].append((A, C))
START, GOAL = map(int, input().split())


# í’€ì´
def solve():
    global N, M, g, START, GOAL

    def bfs(start, weight):
        que = deque([start])
        v = [False] * (N + 1)
        v[start] = True
        while que:
            cur = que.popleft()
            if cur == GOAL:
                return True
            for nxt, limit in g[cur]:
                if weight <= limit and not v[nxt]:
                    que.append(nxt)
                    v[nxt] = True
        return False

    ans = 0
    l = 1
    r = 1_000_000_000
    while l <= r:
        mid = (l + r) // 2
        res = bfs(START, mid)
        if res:
            l = mid + 1
            ans = max(ans, mid)
        else:
            r = mid - 1

    print(ans)
    return


solve()

```
