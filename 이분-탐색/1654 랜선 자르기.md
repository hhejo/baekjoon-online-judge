import sys

input = sys.stdin.readline

K, N = map(int, input().split())
a = [int(input()) for _ in range(K)]


# í’€ì´
def solve():
    global K, N, a

    l = 1
    r = max(a)
    while l <= r:
        mid = (l + r) // 2
        cnt = sum([lan // mid for lan in a])
        if cnt < N:
            r = mid - 1
        else:
            l = mid + 1
    ans = r
    print(ans)
    return


solve()
