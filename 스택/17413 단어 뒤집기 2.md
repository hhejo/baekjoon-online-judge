## CODE 1

```python
ans = ''
S = input().replace('>', '<').split('<')
for i in range(len(S)):
    if i % 2:
        ans += '<' + S[i] + '>'
    else:
        ans += ' '.join(s[::-1] for s in S[i].split())
print(ans)
```

## CODE 2

```python
import sys

input = sys.stdin.readline

s = input().strip()

stk = []
res = []
is_tag = False
for ch in s:
    if ch == '<':
        while stk:
            res.append(stk.pop())
        is_tag = True
        res.append(ch)
    elif ch == '>':
        is_tag = False
        res.append(ch)
    elif ch == ' ':
        while stk:
            res.append(stk.pop())
        res.append(ch)
    else:
        if is_tag:
            res.append(ch)
        else:
            stk.append(ch)
while stk:
    res.append(stk.pop())
ans = ''.join(res)
print(ans)
```
