# 2504 괄호의 값

---

### CODE

```python
def is_valid(input_s):
    stack = []
    for ch in input_s:
        if ch in '([':
            stack.append(ch)
        elif ')' == ch and stack and '(' == stack[-1]:
            stack.pop()
        elif ']' == ch and stack and '[' == stack[-1]:
            stack.pop()
        else:
            return False
    if stack:
        return False
    return True


S = input()
if is_valid(S):
    stk = []
    couple = {')': '(', ']': '['}  # 괄호 짝
    value = {')': 2, ']': 3}  # 괄호의 곱해야 할 값
    for s in S:
        if s in '([':
            stk.append(s)
        elif s in ')]':
            tmp = 0
            while couple[s] != stk[-1]:  # 괄호 짝이면 종료
                tmp += int(stk.pop())  # 닫는 괄호 나올 때까지 숫자 누적
            stk.pop()
            if tmp:  # tmp가 0이 아니면 괄호 중간에 값 있으므로 2나 3을 곱함
                stk.append(tmp * value[s])
            else:  # tmp가 0이면 괄호 중간에 값 없음
                stk.append(value[s])
    print(sum(stk))
else:
    print(0)
```
