# 10799 쇠막대기



```python
stk, cnt = [], 0  # 스택, 쇠막대기 수
for i, p in enumerate(input()):
    if p == '(':  # 여는 괄호면 인덱스 푸시
        stk.append(i)
    else:  # 닫는 괄호면
        if stk[-1] + 1 == i:  # 현재 인덱스와 스택 탑과 차이가 1이면
            stk.pop()  # 레이저이므로 꺼냄
            cnt += len(stk)  # 스택 안의 개수만큼 잘라져서 누적
        else:  # 레이저가 아니라 막대기면
            stk.pop()  # 꺼내고 막대기 끝이므로 1 누적
            cnt += 1
print(cnt)
```
