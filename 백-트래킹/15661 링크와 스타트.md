## CODE 1

```python
import sys

input = sys.stdin.readline

n = int(input())
s = [list(map(int, input().split())) for _ in range(n)]

visited = [False] * n
diff = 1e22


def dfs(depth):
    global n, s, visited, diff
    if depth == n:
        team1 = team2 = 0
        for i in range(n):
            for j in range(n):
                if visited[i] and visited[j]:
                    team1 += s[i][j]
                elif not visited[i] and not visited[j]:
                    team2 += s[i][j]
        diff = min(diff, abs(team1 - team2))
        return
    visited[depth] = True
    dfs(depth + 1)
    visited[depth] = False
    dfs(depth + 1)
    return


dfs(0)
print(diff)
```
