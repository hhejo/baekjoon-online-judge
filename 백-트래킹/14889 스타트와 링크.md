## CODE 1

```python
import sys

input = sys.stdin.readline

n = int(input())
s = [list(map(int, input().split())) for _ in range(n)]

visited = [False] * n
ans = 1e30


def dfs(depth, start):
    global n, s, visited, ans
    if depth == n // 2:
        team1 = team2 = 0
        for i in range(n):
            for j in range(n):
                if visited[i] and visited[j]:
                    team1 += s[i][j]
                elif not visited[i] and not visited[j]:
                    team2 += s[i][j]
        ans = min(ans, abs(team1 - team2))
        return
    for i in range(start, n):
        if not visited[i]:
            visited[i] = True
            dfs(depth + 1, i + 1)
            visited[i] = False
    return


dfs(0, 0)
print(ans)
```

```python
import sys
from itertools import combinations

input = sys.stdin.readline

n = int(input())
s = [list(map(int, input().split())) for _ in range(n)]

ans = 1e10
for r1 in combinations(range(n), n // 2):
    team1 = team2 = 0
    r2 = tuple(set(range(n)) - set(r1))
    for r in combinations(r1, 2):
        team1 += s[r[0]][r[1]]
        team1 += s[r[1]][r[0]]
    for r in combinations(r2, 2):
        team2 += s[r[0]][r[1]]
        team2 += s[r[1]][r[0]]
    ans = min(ans, abs(team1 - team2))

print(ans)
```
