## CODE 1

```python
import sys

input = sys.stdin.readline

n = int(input())

arr = list(range(1, n + 1))
res = [0] * n
visited = [False] * n


def dfs(depth):
    global n, res, visited
    if depth == n:
        print(*res)
        return
    for i in range(n):
        if not visited[i]:
            res[depth] = arr[i]
            visited[i] = True
            dfs(depth + 1)
            visited[i] = False
    return


dfs(0)
```
