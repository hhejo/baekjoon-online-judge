## CODE 1

```python
import sys

input = sys.stdin.readline

n, m = map(int, input().split())
arr = list(map(int, input().split()))
arr.sort()

res = [0] * m


def permutation(n, r, depth, start):
    global arr, res
    if depth == r:
        print(*res)
        return
    for i in range(start, n):
        res[depth] = arr[i]
        permutation(n, r, depth + 1, i + 1)
    return


permutation(n, m, 0, 0)
```
