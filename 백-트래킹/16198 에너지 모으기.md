## CODE 1

```python
import sys

input = sys.stdin.readline

n = int(input())
w = list(map(int, input().split()))

maxx = 0
visited = [False] * n


def dfs(depth, acc):
    global n, w, maxx
    if depth == n - 2:
        maxx = max(maxx, acc)
        return
    for i in range(1, n - 1):
        if not visited[i]:
            left, right = i - 1, i + 1
            while visited[left]:
                left -= 1
            while visited[right]:
                right += 1
            visited[i] = True
            dfs(depth + 1, acc + w[left] * w[right])
            visited[i] = False
    return


dfs(0, 0)
print(maxx)
```

## CODE 2

```python
import sys

input = sys.stdin.readline

n = int(input())
w = list(map(int, input().split()))

maxx = 0


def dfs(depth, acc):
    global n, w, maxx
    if depth == n - 2:
        maxx = max(maxx, acc)
        return
    for i in range(1, len(w) - 1):
        energy = w[i - 1] * w[i + 1]
        bead = w.pop(i)
        dfs(depth + 1, acc + energy)
        w.insert(i, bead)
    return


dfs(0, 0)
print(maxx)
```
