## CODE 1

```python
import sys

input = sys.stdin.readline

n, m = map(int, input().split())

arr = list(range(1, n + 1))
visited = [False] * n
res = [0] * m


def permutation(n, r, k):
    global arr, visited, res
    if k == r:
        print(*res)
        return
    for i in range(n):
        if not visited[i]:
            res[k] = arr[i]
            visited[i] = True
            permutation(n, r, k + 1)
            visited[i] = False
    return


permutation(n, m, 0)
```
