## CODE 1

```python
import sys

input = sys.stdin.readline

n, m = map(int, input().split())
arr = list(map(int, input().split()))
arr.sort()

res = [0] * m
visited = [False] * n


def permutation(n, r, depth):
    global arr, res
    if depth == r:
        print(*res)
        return
    used = 0
    for i in range(n):
        if not visited[i] and used != arr[i]:
            res[depth] = arr[i]
            used = arr[i]
            visited[i] = True
            permutation(n, r, depth + 1)
            visited[i] = False
    return


permutation(n, m, 0)
```
