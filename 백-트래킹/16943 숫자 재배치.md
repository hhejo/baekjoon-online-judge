## CODE 1

```python
import sys
from itertools import permutations

input = sys.stdin.readline

a, b = input().split()

res = []
for perm in permutations(a, len(a)):
    c = ''.join(perm)
    res.append(c)

ans = -1
for c in res:
    if c[0] == '0':
        continue
    if int(c) < int(b):
        ans = max(ans, int(c))

print(ans)
```
