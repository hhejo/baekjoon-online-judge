## CODE 1

```python
import sys

input = sys.stdin.readline

n, m = map(int, input().split())
arr = list(map(int, input().split()))
arr.sort()

visited = [False] * n
res = [0] * m


def permutation(n, r, depth):
    global arr, visited, res
    if depth == r:
        print(*res)
        return
    for i in range(n):
        if not visited[i]:
            res[depth] = arr[i]
            visited[i] = True
            permutation(n, r, depth + 1)
            visited[i] = False
    return


permutation(n, m, 0)
```
