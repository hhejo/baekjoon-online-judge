function solution(n, m, g) {
  const arr = [];
  for (let x = 0; x < n; x++) {
    for (let y = 0; y < m; y++) {
      if (g[x][y] === 1) arr.push([x, y]);
    }
  }

  const v = Array.from(Array(n), () => Array(m).fill(-1));

  dfs();
  function dfs() {
    const [dx, dy] = [
      [-1, 1, 0, 0],
      [0, 0, -1, 1],
    ];
    const que = new Queue();
    for (let [sx, sy] of arr) {
      que.push([sx, sy]);
      v[sx][sy] = 0;
    }
    while (!que.isEmpty()) {
      const [cx, cy] = que.shift();
      for (let i = 0; i < 4; i++) {
        const [nx, ny] = [cx + dx[i], cy + dy[i]];
        if (nx < 0 || n <= nx || ny < 0 || m <= ny) continue;
        if (g[nx][ny] !== 0 || v[nx][ny] !== -1) continue;
        que.push([nx, ny]);
        g[nx][ny] = 1;
        v[nx][ny] = v[cx][cy] + 1;
      }
    }
    return;
  }

  // ì•ˆ ìµì€ í† ë§ˆí†  í™•ì¸
  for (let row = 0; row < n; row++) {
    if (g[row].some((tomato) => tomato === 0)) return console.log(-1);
  }

  }
  return console.log(maxDate);
}

class Node {
  constructor(data) {
    this.data = data;
    this.next = null;
  }
}
class Queue {
  constructor() {
    this.front = null;
    this.rear = null;
    this.length = 0;
  }
  isEmpty() {
    return this.length === 0;
  }
  push(data) {
    const newNode = new Node(data);
    if (this.isEmpty()) this.front = newNode;
    else this.rear.next = newNode;
    this.rear = newNode;
    this.length++;
    return this;
  }
  shift() {
    if (this.isEmpty()) return;
    const { data, next } = this.front;
    this.front = next;
    this.length--;
    return data;
  }
  peek() {
    if (this.isEmpty()) return;
    return this.front.data;
  }
  size() {
    return this.length;
  }
}
