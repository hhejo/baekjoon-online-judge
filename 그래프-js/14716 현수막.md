function solution(m, n, g) {
  const v = new Array(m).fill(0).map(() => new Array(n).fill(0));
  let ans = 0;
  for (let x = 0; x < m; x++) {
    for (let y = 0; y < n; y++) {
      if (g[x][y] === 1 && v[x][y] === 0) {
        dfs(x, y);
        ans += 1;
      }
    }
  }

  function dfs(sx, sy) {
    const stk = [[sx, sy]],
      dx = [-1, -1, 0, 1, 1, 1, 0, -1],
      dy = [0, 1, 1, 1, 0, -1, -1, -1];
    v[sx][sy] = 1;
    while (stk.length > 0) {
      const [cx, cy] = stk.pop();
      for (let i = 0; i < 8; i++) {
        const [nx, ny] = [cx + dx[i], cy + dy[i]];
        if (nx < 0 || m <= nx || ny < 0 || n <= ny) continue;
        if (g[nx][ny] === 0 || v[nx][ny] === 1) continue;
        stk.push([nx, ny]);
        v[nx][ny] = 1;
      }
    }
  }

  console.log(ans);
}

const fs = require("fs");
const filePath = process.platform === "linux" ? "/dev/stdin" : "src/input.txt";
const stdin = fs.readFileSync(filePath).toString().trim().split("\n");
const input = (() => {
  let line = 0;
  return () => stdin[line++];
})();
const [[m, n], g] = [input().split(" ").map(Number), []];
for (let i = 0; i < m; i++) {
  const line = input().split(" ").map(Number);
  g.push(line);
}
solution(m, n, g);
