function solution(n, k) {
  function bfs() {
    let [que, v] = [new Queue(), new Array(100_001).fill(-1)];
    que.enqueue(n);
    v[n] = 0;
    while (que.size() > 0) {
      cur = que.dequeue();
      if (cur === k) return v[cur];
      for (let nxt of [cur - 1, cur + 1, cur * 2]) {
        if (nxt < 0 || nxt > 100_000 || v[nxt] !== -1) continue;
        que.enqueue(nxt);
        v[nxt] = v[cur] + 1;
      }
    }
    return;
