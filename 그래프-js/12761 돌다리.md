function solution(a, b, n, m) {
  const v = new Map(),
    que = new Queue();
  que.enqueue(n);
  v.set(n, 0);
  while (que.size() > 0) {
    const c = que.dequeue();
    if (c === v.has(c)) break;
    const nxts = [c * b, c * a, c + b, c + a, c - b, c - a, c + 1, c - 1];
    for (let nxt of nxts) {
      if (nxt < 0 || 100_000 < nxt || v.has(nxt)) continue;
      que.enqueue(nxt);
      v.set(nxt, v.get(c) + 1);
    }
  }

}

class Node {
  constructor(data) {
    this.data = data;
    this.next = null;
  }
}
class Queue {
  constructor() {
    this.front = null;
    this.rear = null;
    this.length = 0;
  }
  isEmpty() {
    return this.length === 0;
  }
  enqueue(data) {
    const newNode = new Node(data);
    if (this.isEmpty()) this.front = newNode;
    else this.rear.next = newNode;
    this.rear = newNode;
    this.length++;
    return this;
  }
  dequeue() {
    if (this.isEmpty()) return;
    const { data, next } = this.front;
    this.front = next;
    this.length--;
    return data;
  }
  size() {
    return this.length;
  }
}
