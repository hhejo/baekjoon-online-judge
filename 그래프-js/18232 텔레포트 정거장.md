function solution(n, s, e, data) {
  const tels = new Map();
  for (let [x, y] of data) {
    if (tels.has(x)) tels.get(x).push(y);
    else tels.set(x, [y]);
    if (tels.has(y)) tels.get(y).push(x);
    else tels.set(y, [x]);
  }

  const [que, v] = [new Queue(), new Map()];
  que.enqueue(s);
  v.set(s, 0);
  while (que.size() > 0) {
    const cur = que.dequeue();
    if (cur === e) break;
    for (let nxt of tels.get(cur) ?? []) {
      if (v.has(nxt)) continue;
      que.enqueue(nxt);
      v.set(nxt, v.get(cur) + 1);
    }
    for (let nxt of [cur - 1, cur + 1]) {
      if (nxt <= 0 || n < nxt || v.has(nxt)) continue;
      que.enqueue(nxt);
      v.set(nxt, v.get(cur) + 1);
    }
  }

  console.log(v.get(e));
}

class Node {
  constructor(data) {
    this.data = data;
    this.next = null;
  }
}
class Queue {
  constructor() {
    this.front = null;
    this.rear = null;
    this.length = 0;
  }
  isEmpty() {
    return this.length === 0;
  }
  enqueue(data) {
    const newNode = new Node(data);
    if (this.isEmpty()) this.front = newNode;
    else this.rear.next = newNode;
    this.rear = newNode;
    this.length++;
    return this;
  }
  dequeue() {
    if (this.isEmpty()) return;
    const { data, next } = this.front;
    this.front = next;
    this.length--;
    return data;
  }
  size() {
    return this.length;
  }
}

}
