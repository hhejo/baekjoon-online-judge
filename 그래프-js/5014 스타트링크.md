function solution(f, s, g, u, d) {
  const [que, v] = [new Queue(), new Array(f + 1).fill(-1)];
  que.enqueue(s);
  v[s] = 0;
  while (que.size() > 0) {
    const cur = que.dequeue();
    if (cur === g) break;
    for (let nxt of [cur + u, cur - d]) {
      if (nxt < 1 || f < nxt || v[nxt] !== -1) continue;
      que.enqueue(nxt);
      v[nxt] = v[cur] + 1;
    }
  }
  const ans = v[g] === -1 ? "use the stairs" : v[g];
  console.log(ans);
}

class Node {
  constructor(data) {
    this.data = data;
    this.next = null;
  }
}
class Queue {
  constructor() {
    this.front = null;
    this.rear = null;
    this.length = 0;
  }
  isEmpty() {
    return this.length === 0;
  }
  enqueue(data) {
    const newNode = new Node(data);
    if (this.isEmpty()) this.front = newNode;
    else this.rear.next = newNode;
    this.rear = newNode;
    this.length++;
    return this;
  }
  dequeue() {
    if (this.isEmpty()) return;
    const { data, next } = this.front;
    this.front = next;
    this.length--;
    return data;
  }
  size() {
    return this.length;
  }
}
