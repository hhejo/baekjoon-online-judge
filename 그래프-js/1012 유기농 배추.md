function solution(m, n, k, data) {
  let g = new Array(n).fill(0).map((el) => new Array(m).fill(0));
  for (let [y, x] of data) g[x][y] = 1;
  let v = new Array(n).fill(0).map((el) => new Array(m).fill(0));
  let ans = 0;
  for (let x = 0; x < n; x++) {
    for (let y = 0; y < m; y++) {
      if (g[x][y] === 1 && v[x][y] === 0) {
        dfs(x, y);
        ans += 1;
      }
    }
  }
  function dfs(sx, sy) {
    let dx = [-1, 1, 0, 0];
    let dy = [0, 0, -1, 1];
    let stk = [[sx, sy]];
    v[sx][sy] = 1;
    while (stk.length > 0) {
      let [cx, cy] = stk.pop();
      for (let i = 0; i < 4; i++) {
        let [nx, ny] = [cx + dx[i], cy + dy[i]];
        if (nx < 0 || n <= nx || ny < 0 || m <= ny) continue;
        if (g[nx][ny] === 0 || v[nx][ny] === 1) continue;
        stk.push([nx, ny]);
        v[nx][ny] = 1;
      }
    }
  }
  console.log(ans);
}

const fs = require("fs");
const filePath = process.platform === "linux" ? "/dev/stdin" : "src/input.txt";
const stdin = fs.readFileSync(filePath).toString().trim().split("\n");
const input = (() => {
  let line = 0;
  return () => stdin[line++];
})();

let t = +input();
while (t--) {
  let [m, n, k] = input().split(" ").map(Number);
  let data = [];
  for (let i = 0; i < k; i++) {
    let line = input();
    data.push(line.split(" ").map(Number));
  }
  solution(m, n, k, data);
}
