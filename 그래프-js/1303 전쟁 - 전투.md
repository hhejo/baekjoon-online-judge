function solution(n, m, g) {
  let [ally, enemy] = [0, 0],
    v = new Array(m).fill(0).map(() => new Array(n).fill(0));
  for (let x = 0; x < m; x++) {
    for (let y = 0; y < n; y++) {
      if (g[x][y] === "W" && v[x][y] === 0) ally += dfs(x, y, "W") ** 2;
    }
  }

  v = new Array(m).fill(0).map(() => new Array(n).fill(0));
  for (let x = 0; x < m; x++) {
    for (let y = 0; y < n; y++) {
      if (g[x][y] === "B" && v[x][y] === 0) enemy += dfs(x, y, "B") ** 2;
    }
  }

  function dfs(sx, sy, color) {
    const stk = [[sx, sy]],
      dx = [-1, 1, 0, 0],
      dy = [0, 0, -1, 1];
    let cnt = 1;
    v[sx][sy] = 1;
    while (stk.length > 0) {
      const [cx, cy] = stk.pop();
      for (let i = 0; i < 4; i++) {
        const [nx, ny] = [cx + dx[i], cy + dy[i]];
        if (nx < 0 || m <= nx || ny < 0 || n <= ny) continue;
        if (g[nx][ny] !== color || v[nx][ny] === 1) continue;
        stk.push([nx, ny]);
        v[nx][ny] = 1;
        cnt += 1;
      }
    }
    return cnt;
  }

  console.log(ally, enemy);
}

const fs = require("fs");
const filePath = process.platform === "linux" ? "/dev/stdin" : "src/input.txt";
const stdin = fs.readFileSync(filePath).toString().trim().split("\n");
const input = (() => {
  let line = 0;
  return () => stdin[line++];
})();
const [[n, m], g] = [input().split(" ").map(Number), []];
for (let i = 0; i < m; i++) {
  const line = [...input()];
  g.push(line);
}
solution(n, m, g);
