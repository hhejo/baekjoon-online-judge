  const g = new Array(n).fill(0).map(() => new Array(n).fill(0));
  for (let [idx, [x, y]] of data.entries()) g[x - 1][y - 1] = idx + 1;

  const que = new Queue(),
    v = new Array(n).fill(-1).map(() => new Array(n).fill(-1)),
    res = new Array(m + 1).fill(0),
    dx = [-2, -1, 1, 2, 2, 1, -1, -2],
    dy = [1, 2, 2, 1, -1, -2, -2, -1];
  que.enqueue([x - 1, y - 1]);
  v[x - 1][y - 1] = 0;

  while (que.size() > 0) {
    const [cx, cy] = que.dequeue();
    if (g[cx][cy] > 0) res[g[cx][cy]] = v[cx][cy];
    for (let i = 0; i < 8; i++) {
      const [nx, ny] = [cx + dx[i], cy + dy[i]];
      if (nx < 0 || n <= nx || ny < 0 || n <= ny || v[nx][ny] !== -1) continue;
      que.enqueue([nx, ny]);

  }

class Node {
  constructor(data) {
    this.data = data;
    this.next = null;
  }
}
class Queue {
  constructor() {
    this.front = null;
    this.rear = null;
    this.length = 0;
  }
  isEmpty() {
    return this.length === 0;
  }
  enqueue(data) {
    const newNode = new Node(data);
    if (this.isEmpty()) this.front = newNode;
    else this.rear.next = newNode;
    this.rear = newNode;
    this.length++;
    return this;
  }
  dequeue() {
    if (this.isEmpty()) return;
    const { data, next } = this.front;
    this.front = next;
    this.length--;
    return data;
  }
  size() {
    return this.length;
  }
}
