function solution(n, g) {
  let ans = 0;
  for (let h = 0; h < 101; h++) {
    const v = new Array(n).fill(n).map(() => new Array(n).fill(0));
    let cnt = 0;
    for (let x = 0; x < n; x++) {
      for (let y = 0; y < n; y++) {
        if (g[x][y] > h && v[x][y] === 0) {
          dfs(x, y, h, v);
          cnt += 1;
        }
      }
    }
    ans = Math.max(ans, cnt);
  }

  function dfs(sx, sy, h, v) {
    const stk = [[sx, sy]],
      dx = [-1, 1, 0, 0],
      dy = [0, 0, -1, 1];
    v[sx][sy] = 1;
    while (stk.length > 0) {
      const [cx, cy] = stk.pop();
      for (let i = 0; i < 4; i++) {
        const [nx, ny] = [cx + dx[i], cy + dy[i]];
        if (nx < 0 || n <= nx || ny < 0 || n <= ny) continue;
        if (g[nx][ny] <= h || v[nx][ny] === 1) continue;
        stk.push([nx, ny]);
        v[nx][ny] = 1;
      }
    }
  }

  console.log(ans);
}
