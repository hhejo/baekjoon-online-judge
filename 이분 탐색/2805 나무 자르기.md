## CODE

import sys

input = sys.stdin.readline

N, M = map(int, input().split())
a = list(map(int, input().split()))


# í’€ì´
def solve():
    global N, M, a

    def get_max():
        l = 0
        r = max(a)
        while l <= r:
            mid = (l + r) // 2
            cnt = sum([tree - mid if tree - mid > 0 else 0 for tree in a])
            if cnt < M:
                r = mid - 1
            else:
                l = mid + 1
        return r

    ans = get_max()
    print(ans)
    return


solve()
