import sys

input = sys.stdin.readline

N, M = map(int, input().split())
a = list(map(int, input().split()))


# í’€ì´
def solve():
    global N, a

    if N <= M:
        print(N)
        return

    l = 0
    r = 2_000_000_000 * 1_000_000

    while l <= r:
        mid = (l + r) // 2
        start = 0
        end = M
        for i in range(M):
            end += mid // a[i]
        start = end
        for i in range(M):
            if mid % a[i] == 0:
                start -= 1
        start += 1
        if N < start:
            r = mid - 1
        elif N > end:
            l = mid + 1
        else:
            for i in range(M):
                if mid % a[i] == 0:
                    if N == start:
                        print(i + 1)
                        return
                    start += 1
    return


solve()
