## CODE

```python
import sys

input = sys.stdin.readline

X, Y, C = map(float, input().split())


# 풀이
def solve():
    global X, Y, C

    ans = 0
    l = 0
    r = min(X, Y)
    for _ in range(10 ** 5):
        mid = (l + r) / 2.0
        d = mid
        h1 = (X * X - d * d) ** 0.5
        h2 = (Y * Y - d * d) ** 0.5
        h = (h1 * h2) / (h1 + h2)
        if h > C:
            l = mid
        else:
            r = mid

    ans = l
    print(f'{ans:.3f}')
    return


solve()

```
