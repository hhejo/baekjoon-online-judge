# 1920 수 찾기



## 1.

```python
N = int(input())
A = set(map(int, input().split()))
M = int(input())
B = list(map(int, input().split()))
ans = [0] * M
for i in range(M):
    if B[i] in A:
        ans[i] = 1
print(*ans, sep='\n')
```

## 2.

```python
def binary_search(k, l, r):
    global N
    if l > r:
        return 0
    m = (l + r) // 2
    if k == A[m]:
        return 1
    elif k < A[m]:
        return binary_search(k, l, m - 1)
    else:
        return binary_search(k, m + 1, r)


N = int(input())
A = sorted(map(int, input().split()))
M = int(input())
B = list(map(int, input().split()))
ans = [0] * M
for i in range(M):
    ans[i] = binary_search(B[i], 0, N - 1)
print(*ans, sep='\n')
```
