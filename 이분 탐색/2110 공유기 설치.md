## CODE

```python
import sys

input = sys.stdin.readline

N, C = map(int, input().split())
a = [int(input()) for _ in range(N)]


# í’€ì´
def solve():
    global N, C, a

    def possible(length):
        cnt = 1
        last = a[0]
        for home in a:
            if length <= home - last:
                cnt += 1
                last = home
        return cnt >= C

    def binary_search():
        l = 1
        r = a[-1] - a[0]
        res = l
        while l <= r:
            mid = (l + r) // 2
            if possible(mid):
                if res < mid:
                    res = mid
                l = mid + 1
            else:
                r = mid - 1
        return res

    a.sort()
    ans = binary_search()
    print(ans)
    return


solve()
```
