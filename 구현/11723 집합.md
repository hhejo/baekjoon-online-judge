        num |= (1 << x)

        num &= ~(1 << x)
    elif cmd == 'check':
        print(1 if num & (1 << x) else 0)
    elif cmd == 'toggle':
        num ^= (1 << x)
    elif cmd == 'all':
        num = (1 << 21) - 1
    elif cmd == 'empty':
        num = 0
