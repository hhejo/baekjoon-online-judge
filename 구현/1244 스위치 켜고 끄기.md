## CODE 1

```python
import sys

input = sys.stdin.readline

n = int(input())
a = [0] + list(map(int, input().split()))
m = int(input())

conv = [1, 0]

for _ in range(m):
    g, k = map(int, input().split())
    if g == 1:
        for i in range(k, n + 1, k):
            a[i] = conv[a[i]]
    else:
        for i in range(min(k, n + 1 - k)):
            if a[k-i] == a[k+i]:
                a[k-i], a[k+i] = conv[a[k-i]], conv[a[k+i]]
            else:
                break

for i in range(1, n + 1, 20):
    print(*a[i:i+20])
```
