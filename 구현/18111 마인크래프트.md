## CODE 1

```python
import sys

input = sys.stdin.readline

n, m, b = map(int, input().split())
g = [list(map(int, input().split())) for _ in range(n)]

minn, height = int(1e20), 0
for h in range(257):
    take = use = 0
    for x in range(n):
        for y in range(m):
            if g[x][y] > h:
                take += g[x][y] - h
            else:
                use += h - g[x][y]
    if use > take + b:
        continue
    cnt = take * 2 + use
    if minn >= cnt:
        minn, height = cnt, h

print(minn, height)
```
