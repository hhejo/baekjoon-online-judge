## CODE 1

```python
import sys

input = sys.stdin.readline

n, k = map(int, input().split())

arr = list(range(1, n + 1))
res = []

idx = 0
while arr:
    idx = (idx + k - 1) % len(arr)
    res.append(arr.pop(idx))

ans = f'<{", ".join(map(str, res))}>'
print(ans)
```
