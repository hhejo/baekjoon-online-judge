import sys

input = sys.stdin.readline

n = input().strip()

arr = [0] * 10

for num in n:
    arr[int(num)] += 1

tmp = arr[6] + arr[9]
if tmp % 2 == 0:
    arr[6] = arr[9] = tmp // 2
else:
    arr[6] = arr[9] = tmp // 2 + 1

ans = max(arr)

print(ans)
