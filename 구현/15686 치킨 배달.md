## CODE 1

```python
import sys
from itertools import combinations

input = sys.stdin.readline

n, m = map(int, input().split())
g = [list(map(int, input().split())) for _ in range(n)]

chickens = []
houses = []
for x in range(n):
    for y in range(n):
        if g[x][y] == 2:
            chickens.append((x, y))
        elif g[x][y] == 1:
            houses.append((x, y))

ans = int(1e9)
for comb in combinations(chickens, m):
    acc = 0
    for house in houses:
        minn = int(1e9)
        for chicken in comb:
            dist = abs(house[0] - chicken[0]) + abs(house[1] - chicken[1])
            minn = min(minn, dist)
        acc += minn
    ans = min(ans, acc)

print(ans)
```
