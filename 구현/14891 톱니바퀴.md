```python
import sys

input = sys.stdin.readline

g = [list(map(int, input().strip())) for _ in range(4)]
k = int(input())
a = [list(map(int, input().split())) for _ in range(k)]

g.insert(0, [])

for num, direction in a:
    rotation = [0] * 5
    rotation[num] = direction

    cur = num
    left = num - 1
    while left >= 1:
        if g[cur][6] == g[left][2]:
            break
        value = 1
        if rotation[cur] == 1:
            value = -1
        rotation[left] = value
        cur = left
        left -= 1

    cur = num
    right = num + 1
    while right <= 4:
        if g[cur][2] == g[right][6]:
            break
        value = 1
        if rotation[cur] == 1:
            value = -1
        rotation[right] = value
        cur = right
        right += 1

    for i in range(1, 5):
        if rotation[i] == 0:
            continue
        tmp = [0] * 8
        if rotation[i] == -1:
            for j in range(7):
                tmp[j] = g[i][j + 1]
            tmp[7] = g[i][0]
        else:
            for j in range(1, 8):
                tmp[j] = g[i][j - 1]
            tmp[0] = g[i][7]
        for j in range(8):
            g[i][j] = tmp[j]

ans = 0
for i in range(1, 5):
    ans += g[i][0] * (2 ** (i - 1))
print(ans)
```
