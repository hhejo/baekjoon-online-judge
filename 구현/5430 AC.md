```python
import sys

input = sys.stdin.readline

for test_case in range(int(input())):
    p = input().strip()
    n = int(input())
    x = input().strip()
    arr = []
    if x == '[]':
        arr = []
    else:
        arr = list(map(int, x[1:-1].split(',')))

    front, rear = 0, len(arr)
    reverse = False
    ans = 'error'
    for cmd in p:
        if cmd == 'R':
            reverse = not reverse
        else:
            if abs(front - rear) == 0:
                break
            else:
                if reverse:
                    rear -= 1
                else:
                    front += 1
    else:
        if reverse:
            ans = f'[{",".join(map(str, reversed(arr[front:rear])))}]'
        else:
            ans = f'[{",".join(map(str, arr[front:rear]))}]'
    print(ans)
```
