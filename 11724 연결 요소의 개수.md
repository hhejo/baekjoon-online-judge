                        visited[v] = i
    return len(set(visited[1:]))


N, M = map(int, input().split())

for _ in range(M):

    gr[V].append(U)
ans = dfs()

```python
import sys

input = sys.stdin.readline


def dfs():
    global N, gr
    visited = [0] * (N + 1)
    for i in range(1, N + 1):
        if not visited[i]:
            s = [i]
            visited[i] = i
            while s:
                c = s.pop()
                for j in range(1, N + 1):
                    if gr[c][j] and not visited[j]:
                        s.append(j)
                        visited[j] = i
    return len(set(visited[1:]))


N, M = map(int, input().split())
gr = [[0] * (N + 1) for _ in range(N + 1)]
for _ in range(M):
    U, V = map(int, input().split())
    gr[U][V] = gr[V][U] = 1
ans = dfs()
print(ans)
```
