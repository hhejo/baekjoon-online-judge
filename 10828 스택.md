# 10828 스택



```python
import sys

N = int(sys.stdin.readline())
stk = []
for cmd in [sys.stdin.readline().rstrip() for _ in range(N)]:
    if 'push' in cmd:
        stk.append(int(cmd.split()[-1]))
    elif 'pop' in cmd:
        if stk:
            print(stk.pop())
        else:
            print(-1)
    elif 'size' in cmd:
        print(len(stk))
    elif 'empty' in cmd:
        print(0 if len(stk) else 1)
    elif 'top' in cmd:
        if stk:
            print(stk[-1])
        else:
            print(-1)
```
