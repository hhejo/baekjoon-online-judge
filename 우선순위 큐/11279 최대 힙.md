def push_heap(x):
    global heap, idx
    idx += 1
    heap[idx] = x
    i = idx
    while i > 1:
        p = i // 2
        if heap[p] < heap[i]:
            heap[p], heap[i] = heap[i], heap[p]
            i //= 2
        else:
            break
    return


def pop_heap():
    global heap, idx
    if idx == 0:
        return 0
    res = heap[1]
    heap[1] = heap[idx]
    heap[idx] = 0
    idx -= 1
    i = 1
    while i * 2 <= idx:
        l = i * 2
        r = i * 2 + 1
        if heap[i] > heap[l] and heap[i] > heap[r]:
            break
        elif heap[l] > heap[r]:
            heap[l], heap[i] = heap[i], heap[l]
            i = l
        else:
            heap[r], heap[i] = heap[i], heap[r]
            i = r
    return res
