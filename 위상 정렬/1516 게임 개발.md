import sys
from collections import deque

input = sys.stdin.readline

#


def topology_sort():
    global N, g, ind, w
    que = deque()
    dp = [0] * (N + 1)
    for i in range(1, N + 1):
        if ind[i] == 0:
            que.append(i)
            dp[i] = w[i]
    while que:
        cur = que.popleft()
        for nxt in g[cur]:
            ind[nxt] -= 1
            dp[nxt] = max(dp[nxt], dp[cur] + w[nxt])
            if ind[nxt] == 0:
                que.append(nxt)
    return max(dp)


N = int(input())
g = [[] for _ in range(N + 1)]
ind = [0] * (N + 1)
w = [0] * (N + 1)
for i in range(1, N + 1):
    A = list(map(int, input().split()))
    w[i] = A[0]
    for j in range(A[1]):
        g[A[j + 2]].append(i)
        ind[i] += 1

ans = topology_sort()
print(ans)
