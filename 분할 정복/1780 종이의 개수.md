# 1780 종이의 개수

## CODE

```python
import sys

input = sys.stdin.readline

N = int(input())
a = [list(map(int, input().split())) for _ in range(N)]


# 풀이
def solve():
    global N, a

    # (x, y)부터 n 크기의 정사각형 안의 모든 값이 같은지 확인하는 함수
    def check(x, y, n):
        for i in range(x, x + n):
            for j in range(y, y + n):
                if a[x][y] != a[i][j]:
                    return False
        return True

    # 9 조각으로 분할하는 함수
    def divide(x, y, n):
        # 영역 내 모든 값이 같음
        if check(x, y, n):
            cnt[a[x][y] + 1] += 1  # 해당 종이 개수 추가
            return
        m = n // 3  # 3등분
        # 분할 재귀
        for i in range(3):
            for j in range(3):
                divide(x + m * i, y + m * j, m)
        return

    cnt = [0, 0, 0]  # -1, 0, 1으로만 이루어진 종이 개수
    divide(0, 0, N)
    print(*cnt, end='\n')  # 정답 출력
    return


solve()

```
