## CODE

```python
import sys

input = sys.stdin.readline
sys.setrecursionlimit(10 ** 6)

N, r, c = map(int, input().split())


# 풀이
def solve():
    global N, r, c

    def z(n, r, c):
        if n == 0:
            return 0
        return 2 * (r % 2) + (c % 2) + 4 * z(n - 1, r // 2, c // 2)

    ans = z(N, r, c)
    print(ans)
    return


solve()

```
