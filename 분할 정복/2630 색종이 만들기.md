import sys

input = sys.stdin.readline

N = int(input())
a = [list(map(int, input().split())) for _ in range(N)]


def solve():
    global N, a

    def all_same(x, y, n):
        for i in range(n):
            for j in range(n):
                if a[x][y] != a[x + i][y + j]:
                    return False
        return True

    def divide(x, y, n):
        if all_same(x, y, n):
            res[a[x][y]] += 1
            return

        divide(x, y, m)
        divide(x + m, y, m)
        divide(x, y + m, m)
        divide(x + m, y + m, m)
        return

    res = [0, 0]
    divide(0, 0, N)
    print(*res, sep='\n')
