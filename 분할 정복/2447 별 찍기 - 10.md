## CODE

```python
import sys

input = sys.stdin.readline
sys.setrecursionlimit(10 ** 6)

N = int(input())


# 풀이
def solve():
    global N
    STAR = 1
    BLANK = 0
    a = [[''] * N for _ in range(N)]

    def star(x, y, n, ch):
        if ch == BLANK:
            for i in range(x, x + n):
                for j in range(y, y + n):
                    a[i][j] = ' '
            return

        if n == 1:
            a[x][y] = '*'
            return

        new_ch = STAR
        m = n // 3
        for i in range(3):
            for j in range(3):
                if i == 1 and j == 1:
                    new_ch = BLANK
                else:
                    new_ch = STAR
                star(x + i * m, y + j * m, m, new_ch)
        return

    star(0, 0, N, STAR)

    for row in a:
        print(''.join(map(str, row)))
    return


solve()

```
