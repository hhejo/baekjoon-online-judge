import sys

input = sys.stdin.readline

#

N = int(input())
a = [list(map(int, input().split())) for _ in range(N)]

a.append(a[0])
x = y = 0
for i in range(N):
    x += a[i][0] * a[i + 1][1]
    y += a[i][1] * a[i + 1][0]
ans = abs((x - y) / 2)
print(f'{ans:.1f}')
