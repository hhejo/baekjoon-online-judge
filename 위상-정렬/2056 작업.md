import sys
from collections import deque

input = sys.stdin.readline

#


def topology_sort():
    global N, g, w, ind
    que = deque()
    dp = [0] * (N + 1)
    for i in range(1, N + 1):
        if ind[i] == 0:
            que.append(i)
            dp[i] = w[i]
    while que:
        cur = que.popleft()
        for nxt in g[cur]:
            ind[nxt] -= 1
            dp[nxt] = max(dp[nxt], dp[cur] + w[nxt])
            if ind[nxt] == 0:
                que.append(nxt)
    return dp


N = int(input())
g = [[] for _ in range(N + 1)]
w = [0] * (N + 1)
ind = [0] * (N + 1)
for i in range(1, N + 1):
    C, *A = list(map(int, input().split()))
    w[i] = C
    for j in range(len(A) - 1):
        g[A[j]].append(i)
        ind[i] += 1

res = topology_sort()
print(*res[1:], sep='\n')
