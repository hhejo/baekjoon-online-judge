## CODE

import sys
from collections import deque

input = sys.stdin.readline

for test_case in range(int(input())):
    N = int(input())
    a = list(input().rstrip().split())
    que = deque([a[0]])
    for i in range(1, N):
        if a[i] <= que[0]:
            que.appendleft(a[i])
        else:
            que.append(a[i])
    print(''.join(que))
