import sys
from collections import deque

que = deque()
N = int(sys.stdin.readline())
for cmd in sys.stdin:
    if 'push' in cmd:
        que.append(cmd.split()[1])
    elif 'pop' in cmd:
        if que:
            print(que.popleft())
        else:
            print(-1)
    elif 'size' in cmd:
        print(len(que))
    elif 'empty' in cmd:
        if que:
            print(0)
        else:
            print(1)
    elif 'front' in cmd:
        if que:
            print(que[0])
        else:
            print(-1)
    elif 'back' in cmd:
        if que:
            print(que[-1])
        else:
            print(-1)
