import sys

input = sys.stdin.readline

E, S, M = map(int, input().split())


def solve():
    global E, S, M

    e = s = m = 1
    cnt = 1
    while True:
        if e == E and s == S and m == M:
            break
        e = e % 15 + 1
        s = s % 28 + 1
        m = m % 19 + 1
        cnt += 1

    print(cnt)
    return


solve()
