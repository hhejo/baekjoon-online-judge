    stk.push(s);
    continue;

    flag = true;
    break;
  }
  if (s === ")") {
    if (stk.at(-1) === "(") {

      flag = true;
      break;

    stk.pop();
    stk.push(acc * 2);

      stk.pop();
      stk.push(3);

    while (stk.length > 0 && isFinite(stk.at(-1))) acc += stk.pop();
    if (stk.length === 0 || stk.at(-1) !== "[") {
      flag = true;
      break;
    }
    stk.pop();
    stk.push(acc * 3);
  }
