import sys
from collections import deque

input = sys.stdin.readline

N = int(input())
g = [list(map(int, input().split())) for _ in range(N)]

sx = sy = 0
for i in range(N):
    for j in range(N):
        if g[i][j] == 9:
            sx, sy = i, j

shark = 2
eat = 0
sec = 0


def bfs(sx, sy):
    global N, g
    dx, dy = (0, -1, 0, 1), (-1, 0, 1, 0)
    que = deque([(sx, sy, 0, 0)])
    v = [[False] * N for _ in range(N)]
    v[sx][sy] = True
    fish = []
    while que:
        cx, cy, dist, size = que.popleft()
        if size > 0 and size < shark:
            fish.append((cx, cy, dist, size))
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < N and 0 <= ny < N and shark >= g[nx][ny] and not v[nx][ny]:
                que.append((nx, ny, dist + 1, g[nx][ny]))
                v[nx][ny] = True
    return fish


def get_target_fish(fish):
    target = None
    target = sorted(fish, key=lambda k: (k[2], k[0], k[1]))[0]
    return target


def go_and_eat(target):
    global sec, eat, shark, sx, sy, g
    sec += target[2]
    eat += 1
    if eat >= shark:
        shark += 1
        eat = 0
    g[sx][sy] = 0
    g[target[0]][target[1]] = 9
    sx, sy = target[0], target[1]
    return


while True:
    fish = bfs(sx, sy)
    if not fish:
        break
    target = get_target_fish(fish)
    go_and_eat(target)


print(sec)

```

import sys
from collections import deque

input = sys.stdin.readline

n = int(input())
g = [list(map(int, input().split())) for _ in range(n)]

size, eat, sec = 2, 0, 0
x = y = 0
for i in range(n):
    for j in range(n):
        if g[i][j] == 9:
            x, y = i, j
            break


def bfs(sx, sy):
    global n, g, size
    dx, dy = (0, -1, 0, 1), (-1, 0, 1, 0)
    visited = [[-1] * n for _ in range(n)]
    visited[sx][sy] = 0
    que = deque([(sx, sy, 0)])
    res = []
    while que:
        cx, cy, fish_size = que.popleft()
        if 0 < fish_size < size:
            res.append((cx, cy, visited[cx][cy]))
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < n and 0 <= ny < n and visited[nx][ny] == -1 and g[nx][ny] <= size:
                visited[nx][ny] = visited[cx][cy] + 1
                que.append((nx, ny, g[nx][ny]))
    res.sort(key=lambda x: (x[2], x[0], x[1]))
    return res


while True:
    res = bfs(x, y)
    if not res:
        break
    nx, ny, dist = res[0]
    eat += 1
    if eat == size:
        size += 1
        eat = 0
    sec += dist
    g[x][y], g[nx][ny] = 0, 9
    x, y = nx, ny

print(sec)
```
