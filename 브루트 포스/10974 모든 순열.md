import sys

input = sys.stdin.readline

n = int(input())


def permutation(n, depth):
    global visited
    if depth == n:
        print(*res)
        return
    for i in range(n):
        if not visited[i]:
            res[depth] = arr[i]
            visited[i] = True
            permutation(n, depth + 1)
            visited[i] = False
    return


arr = list(range(1, n + 1))
res = [0] * n
visited = [False] * n
permutation(n, 0)
