MAP = [list(input()) for _ in range(N)]
ans = 2500
W = [['W', 'B', 'W', 'B', 'W', 'B', 'W', 'B'],
     ['B', 'W', 'B', 'W', 'B', 'W', 'B', 'W'],

     ['W', 'B', 'W', 'B', 'W', 'B', 'W', 'B'],
     ['B', 'W', 'B', 'W', 'B', 'W', 'B', 'W']]
B = [['B', 'W', 'B', 'W', 'B', 'W', 'B', 'W'],
     ['W', 'B', 'W', 'B', 'W', 'B', 'W', 'B'],

     ['B', 'W', 'B', 'W', 'B', 'W', 'B', 'W'],
     ['W', 'B', 'W', 'B', 'W', 'B', 'W', 'B'],
     ['B', 'W', 'B', 'W', 'B', 'W', 'B', 'W'],
     ['W', 'B', 'W', 'B', 'W', 'B', 'W', 'B'],]
for i in range(0, N - 7):

        cnt = 0

            for y in range(8):
                if B[x][y] != MAP[i+x][j+y]:
                    cnt += 1
        ans = min(ans, cnt)
print(ans)

## 2.

```python
N, M = map(int, input().split())
MAP = [list(input()) for _ in range(N)]
ans = 2500
W = (('WB' * 4, 'BW' * 4) * 4)
B = (('BW' * 4, 'WB' * 4) * 4)
for i in range(0, N - 7):
    for j in range(0, M - 7):
        cnt_W = cnt_B = 0
        for x in range(8):
            for y in range(8):
                if W[x][y] != MAP[i+x][j+y]:
                    cnt_W += 1
                if B[x][y] != MAP[i+x][j+y]:
                    cnt_B += 1
        ans = min(ans, cnt_W, cnt_B)
print(ans)
```
