def check(arr, _max):
    for i in range(N):
        for j in range(N - 1):
            arr[i][j], arr[i][j + 1] = arr[i][j + 1], arr[i][j]

            cnt = 1
            for k in range(N - 1):
                if arr[i][k] == arr[i][k + 1]:
                    cnt += 1
                    _max = max(_max, cnt)
                else:
                    cnt = 1

            cnt = 1
            for k in range(N - 1):
                if arr[k][j] == arr[k + 1][j]:
                    cnt += 1
                    _max = max(_max, cnt)
                else:
                    cnt = 1

            cnt = 1
            for k in range(N - 1):
                if arr[k][j + 1] == arr[k + 1][j + 1]:
                    cnt += 1
                    _max = max(_max, cnt)
                else:
                    cnt = 1

            arr[i][j], arr[i][j + 1] = arr[i][j + 1], arr[i][j]
    return _max


N = int(input())
ARR = [list(input()) for _ in range(N)]
ans = 1
ans = check(ARR, ans)
ans = check(list(map(list, zip(*ARR))), ans)
print(ans)

def check():
    global arr
    _max = 1
    for i in range(N):
        cnt = 1
        for j in range(1, N):
            if arr[i][j] == arr[i][j-1]:
                cnt += 1
                _max = max(_max, cnt)
            else:
                cnt = 1
        cnt = 1
        for j in range(1, N):
            if arr[j][i] == arr[j-1][i]:
                cnt += 1
                _max = max(_max, cnt)
            else:
                cnt = 1
    return _max


N = int(input())
arr = [list(input()) for _ in range(N)]
ans = 0
for i in range(N):
    for j in range(N):
        if j < N - 1:
            arr[i][j], arr[i][j+1] = arr[i][j+1], arr[i][j]
            ans = max(ans, check())
            arr[i][j], arr[i][j+1] = arr[i][j+1], arr[i][j]
        if i < N - 1:
            arr[i][j], arr[i+1][j] = arr[i+1][j], arr[i][j]
            ans = max(ans, check())
            arr[i][j], arr[i+1][j] = arr[i+1][j], arr[i][j]
print(ans)
