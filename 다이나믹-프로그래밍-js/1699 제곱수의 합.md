function solution(n) {
  const dp = new Array(n + 1).fill(0).map((v, i) => i);
  for (let i = 1; i <= n; i++) {
    for (let j = 1; j * j <= i; j++) {
      if (dp[i] > dp[i - j * j] + 1) {
        dp[i] = dp[i - j * j] + 1;
      }
    }
  }
  const ans = dp[n];
  console.log(ans);
}
