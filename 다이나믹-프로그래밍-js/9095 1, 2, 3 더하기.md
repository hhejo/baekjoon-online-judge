function solution(n) {
  const dp = new Array(n + 1).fill(0);
  dp[1] = 1;
  if (n >= 2) dp[2] = 2;
  if (n >= 3) dp[3] = 4;
  for (let i = 4; i <= n; i++) {
    dp[i] = dp[i - 1] + dp[i - 2] + dp[i - 3];
  }
  const ans = dp[n];
  console.log(ans);
}
