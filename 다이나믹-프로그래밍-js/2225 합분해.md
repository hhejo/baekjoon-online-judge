function solution(n, k) {
  const [dp, mod] = [[], 1_000_000_000];
  for (let i = 0; i <= k; i++) dp[i] = new Array(n + 1);
  for (let i = 0; i <= n; i++) dp[1][i] = 1;
  for (let i = 2; i <= k; i++) {
    for (let j = 0; j <= n; j++) {
      if (j === 0) dp[i][j] = 1;
      else dp[i][j] = (dp[i][j - 1] + dp[i - 1][j]) % mod;
    }
  }
  const ans = dp[k][n];
  return console.log(ans);
}
