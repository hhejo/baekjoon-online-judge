function solution(n) {
  const dp = new Array(n + 1).fill(0).map(() => new Array(10).fill(0));
  const mod = 10_007;
  for (let i = 0; i <= 9; i++) dp[1][i] = 1;
  for (let x = 2; x <= n; x++) {
    for (let y = 0; y <= 9; y++) {
      for (let z = 0; z <= y; z++) {
        dp[x][y] += dp[x - 1][z] % mod;
      }
    }
  }
  const ans = dp[n].reduce((acc, cur) => acc + cur, 0) % mod;
  console.log(ans);
}
