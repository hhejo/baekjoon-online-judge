function solution(n, arr) {
  const dp = new Array(n).fill(0);
  const { max } = Math;
  dp[0] = arr[0];
  if (n >= 1) dp[1] = arr[0] + arr[1];
  if (n >= 2) dp[2] = max(arr[1] + arr[2], arr[0] + arr[2]);
  for (let i = 3; i < n; i++) {
    dp[i] = max(arr[i] + dp[i - 2], arr[i] + arr[i - 1] + dp[i - 3]);
  }
  const ans = dp[n - 1];
  console.log(ans);
}
