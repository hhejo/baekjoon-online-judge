function solution(n, arr) {
  const dp = new Array(n + 1).fill(0);
  const { max } = Math;
  dp[1] = arr[1];
  if (n >= 2) dp[2] = arr[1] + arr[2];
  for (let i = 3; i <= n; i++) {
    dp[i] = max(dp[i - 1], dp[i - 2] + arr[i], dp[i - 3] + arr[i - 1] + arr[i]);
  }
  const ans = dp[n];
  console.log(ans);
}
