import sys
from collections import deque

N = int(sys.stdin.readline())
que = deque()
for line in sys.stdin:
    cmd = line.split()
    if 'push' in cmd:
        que.append(cmd[1])
    elif 'pop' in cmd:
        print(que.popleft() if que else -1)
    elif 'size' in cmd:
        print(len(que))
    elif 'empty' in cmd:
        print(0 if que else 1)
    elif 'front' in cmd:
        print(que[0] if que else -1)
    elif 'back' in cmd:
        print(que[-1] if que else -1)
