## CODE 1

```python
import sys

input = sys.stdin.readline

N = int(input())
g = [list(map(int, input().split())) for _ in range(N)]

for k in range(N):
    for a in range(N):
        for b in range(N):
            if g[a][k] and g[k][b]:
                g[a][b] = 1

for r in g:
    print(*r)

```

import sys

N = int(input())
MAP = [list(map(int, input().split())) for _ in range(N)]

for k in range(N):
    for a in range(N):
        for b in range(N):
            if MAP[a][b] == 1 or (MAP[a][k] == MAP[k][b] == 1):
                MAP[a][b] = 1

print(*[' '.join(map(str, row)) for row in MAP], sep='\n')
