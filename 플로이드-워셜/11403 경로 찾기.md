## CODE

import sys

input = sys.stdin.readline

N = int(input())
g = [list(map(int, input().split())) for _ in range(N)]

for k in range(N):
    for a in range(N):
        for b in range(N):
            if g[a][k] and g[k][b]:
                g[a][b] = 1

for r in g:
    print(*r)
