import sys

input = sys.stdin.readline

#

N = int(input())
D = [list(map(int, input().split())) for _ in range(N)]
road = [[True] * N for _ in range(N)]

res = 0
for k in range(N):
    for a in range(N):
        for b in range(N):
            if k == a or a == b or b == k:
                continue
            if road[a][b] and road[a][k] and road[k][b]:
                if D[a][b] == D[a][k] + D[k][b]:
                    road[a][b] = False
                elif D[a][b] > D[a][k] + D[k][b]:
                    res = -1

if res != -1:
    for a in range(N):
        for b in range(a, N):
            if road[a][b]:
                res += D[a][b]
print(res)
