## CODE 1

```python
import sys

input = sys.stdin.readline

N, M = map(int, input().split())
INF = int(1e9)
g = [[INF if i != j else 0 for i in range(N)] for j in range(N)]
for _ in range(M):
    A, B = map(int, input().split())
    g[A - 1][B - 1] = g[B - 1][A - 1] = 1

for k in range(N):
    for a in range(N):
        for b in range(N):
            g[a][b] = min(g[a][b], g[a][k] + g[k][b])

sums = [0] * N
for i, r in enumerate(g):
    sums[i] = sum(r)

min_value = INF
ans = 0
for i in range(N):
    if min_value > sums[i]:
        min_value = sums[i]
        ans = i + 1
print(ans)

```
