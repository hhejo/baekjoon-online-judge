## CODE 1

```python
import sys

input = sys.stdin.readline

#

N, M = int(input()), int(input())
INF = int(1e9)
d = [[INF if i != j else 0 for j in range(N)] for i in range(N)]
for _ in range(M):
    A, B, C = map(int, input().split())
    if d[A - 1][B - 1] > C:
        d[A - 1][B - 1] = C

for k in range(N):
    for a in range(N):
        for b in range(N):
            d[a][b] = min(d[a][b], d[a][k] + d[k][b])

for i in range(N):
    for j in range(N):
        if d[i][j] == INF:
            d[i][j] = 0
print(*[' '.join(map(str, row)) for row in d], sep='\n')

```
