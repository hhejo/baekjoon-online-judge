## CODE 1

```python
import sys

input = sys.stdin.readline

#

N = int(input())
arr = list(map(int, input().split()))

ans = 2000000000
al = ar = 0
for i in range(N - 1):
    l, r = i + 1, N - 1
    while l <= r:
        mid = (l + r) // 2
        summ = arr[i] + arr[mid]
        if abs(summ) < ans:
            ans = abs(summ)
            al, ar = i, mid
            if summ == 0:
                break
        if summ < 0:
            l = mid + 1
        else:
            r = mid - 1

print(arr[al], arr[ar])

```
