import sys

input = sys.stdin.readline


def find(x):
    global p
    if x == p[x]:
        return x
    p[x] = find(p[x])
    return p[x]


def union(x, y):
    global p
    x = find(x)
    y = find(y)
    if x != y:
        p[y] = x
    return


N, M = map(int, input().split())
p = [i for i in range(N + 1)]
for _ in range(M):
    K, A, B = map(int, input().split())
    if K == 0:
        union(A, B)
    else:
        if find(A) == find(B):
            print('YES')
        else:
            print('NO')
