```python
import sys

input = sys.stdin.readline

#


def find(x):
    global p
    if p[x] != x:
        p[x] = find(p[x])
    return p[x]


def union(x, y):
    global p
    x, y = find(x), find(y)
    if x != y:
        p[y] = x
    return


N, M = map(int, input().split())
p = [i for i in range(N)]
ans = 0
for i in range(1, M + 1):
    A, B = map(int, input().split())
    if find(A) == find(B):
        if ans == 0:
            ans = i
        continue
    union(A, B)

print(ans)

```
