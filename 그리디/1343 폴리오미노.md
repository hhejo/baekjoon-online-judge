import sys
#############################
sys.stdin = open('input.txt')
#############################
input = sys.stdin.readline

s = input()

ans = ''
cnt = 0
for ch in s:
    if ch == 'X':
        cnt += 1
        if cnt == 4:
            ans += 'AAAA'
            cnt = 0
    elif ch == '.':
        if cnt == 1 or cnt == 3:
            ans = '-1'
            break
        elif cnt == 4:
            ans += 'AAAA'
        elif cnt == 2:
            ans += 'BB'
        ans += '.'
        cnt = 0

if cnt == 4:
    ans += 'AAAA'
elif cnt == 2:
    ans += 'BB'
elif cnt == 1 or cnt == 3:
    ans = '-1'

print(ans)
