## CODE 1

```python
import sys

input = sys.stdin.readline

n, l, k = map(int, input().split())
a = [tuple(map(int, input().split())) for _ in range(n)]
a.sort(key=lambda x: x[1])

acc = cnt = 0
for i in range(n):
    if cnt == k:
        break
    if a[i][1] <= l:
        acc += 140
        cnt += 1
    elif a[i][0] <= l:
        acc += 100
        cnt += 1

print(acc)
```
