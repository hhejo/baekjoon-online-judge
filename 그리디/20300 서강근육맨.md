import sys

input = sys.stdin.readline

n = int(input())

t.sort()

last = 0
if len(t) % 2 == 1:
    last = t.pop()
res = [t[i] + t[-i - 1] for i in range(len(t) // 2)]

ans = max(last, *res)
print(ans)
