import sys

input = sys.stdin.readline

t = int(input())

counts = [0] * 3
seconds = [300, 60, 10]
for i in range(3):
    if t >= seconds[i]:
        counts[i] += t // seconds[i]
        t %= seconds[i]
if t > 0:
    print(-1)
else:
    print(*counts)
