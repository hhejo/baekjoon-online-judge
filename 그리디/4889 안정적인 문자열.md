## CODE 1

```python
import sys

input = sys.stdin.readline

test_case = 0
while True:
    test_case += 1
    s = input().strip()
    if '-' in s:
        break
    cnt, stk = 0, []
    for ch in s:
        if ch == '{':
            stk.append(ch)
        else:
            if stk:
                stk.pop()
            else:
                cnt += 1
                stk.append('{')
    cnt += len(stk) // 2
    print(f'{test_case}. {cnt}')
```
