## CODE 1

import sys

input = sys.stdin.readline

for test_case in range(int(input())):
    n = int(input())
    s = input().strip()
    g = input().strip()

    arr = []
    for i in range(n):
        if s[i] != g[i]:
            arr.append(s[i])

    ans = 0
    if arr.count('B') >= arr.count('W'):
        ans = arr.count('B')
    else:
        ans = arr.count('W')

    print(ans)
