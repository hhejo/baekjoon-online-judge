## CODE 1

```python
import sys

input = sys.stdin.readline

n = int(input())
a = [int(input()) for _ in range(n)]

maxx = max(a)
max_idx = a.index(maxx)
cnt = len(a) - max_idx - 1
prev = maxx
for x in reversed(a[:max_idx]):
    if x == prev - 1:
        prev = x
    else:
        cnt += 1

print(cnt)
```
