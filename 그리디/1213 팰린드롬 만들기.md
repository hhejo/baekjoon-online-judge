## CODE 1

import sys
from collections import Counter

input = sys.stdin.readline

s = input().strip()

arr = sorted(s)
counters = Counter(arr)

odd, mid, ans = 0, '', ''
for ch, cnt in counters.items():
    ans += ch * (cnt // 2)
    if cnt % 2 == 1:
        odd += 1
        mid = ch

ans = ans + mid + ans[::-1]
if odd > 1:
    ans = "I'm Sorry Hansoo"
print(ans)
