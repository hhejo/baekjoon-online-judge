## CODE 1

```python
import sys

#############################
sys.stdin = open('input.txt')
#############################

input = sys.stdin.readline

N = int(input())
arr = [int(input()) for _ in range(N)]


# 풀이
def solve():
    global N, arr

    ps = []
    ns = []
    one = 0
    zero = 0
    minus = 0

    for num in arr:
        if num == 0:
            zero += 1
        elif num == 1:
            one += 1
        elif num == -1:
            minus += 1
        elif num > 1:
            ps.append(num)
        elif num < -1:
            ns.append(num)

    ps.sort()
    ns.sort(reverse=True)

    ans = 0

    if len(ps) % 2 == 0:
        ans += sum([ps[i] * ps[i + 1] for i in range(0, len(ps), 2)])
    else:
        ans += sum([ps[i] * ps[i + 1] for i in range(1, len(ps), 2)]) + ps[0]

    if len(ns) % 2 == 0:
        ans += sum([ns[i] * ns[i + 1] for i in range(0, len(ns), 2)])
    else:
        ans += sum([ns[i] * ns[i + 1] for i in range(1, len(ns), 2)])
        if minus > 0:
            ans += -1 * ns[0]
            minus -= 1
        else:
            if zero > 0:
                ans += 0 * ns[0]
                zero -= 1
            else:
                ans += ns[0]

    ans += minus // 2
    minus %= 2
    if minus == 1:
        if zero > 0:
            ans += -1 * 0
            zero -= 1
            minus = 0
        else:
            ans -= 1
            minus = 0

    ans += one

    print(ans)
    return


solve()

```

## 해설 1

1은 더하는 것이 이득이다

-1은 곱하는 것이 이득이다

0은 -1이 홀수개이면 -1에 곱해준다

`31256KB`, `44ms`

## CODE 2

```python
import sys

input = sys.stdin.readline

N = int(input())
arr = [int(input()) for _ in range(N)]


# 풀이
def solve():
    global N, arr
    ps = []
    ns = []
    ans = 0

    for num in arr:
        if num == 1:
            ans += 1
        elif num > 1:
            ps.append(num)
        else:
            ns.append(num)

    ps.sort(reverse=True)
    ns.sort()

    if len(ps) % 2 == 0:
        for i in range(0, len(ps), 2):
            ans += ps[i] * ps[i + 1]
    else:
        for i in range(0, len(ps) - 1, 2):
            ans += ps[i] * ps[i + 1]
        ans += ps[-1]

    if len(ns) % 2 == 0:
        for i in range(0, len(ns), 2):
            ans += ns[i] * ns[i + 1]
    else:
        for i in range(0, len(ns) - 1, 2):
            ans += ns[i] * ns[i + 1]
        ans += ns[-1]

    print(ans)
    return


solve()

```

## 해설 2

양수 모음은 1 이상인 수들을 모은다

음수 모음은 0, -1도 다 포함한다

`31256KB`, `56ms`
