    ans = 0

        if num == 1:
            ans += 1
        elif num > 1:
            ps.append(num)
        else:

            ans += ps[i] * ps[i + 1]
        ans += ps[-1]

    if len(ns) % 2 == 0:
        for i in range(0, len(ns), 2):
            ans += ns[i] * ns[i + 1]
    else:
        for i in range(0, len(ns) - 1, 2):
            ans += ns[i] * ns[i + 1]
        ans += ns[-1]

    print(ans)
    return


solve()
