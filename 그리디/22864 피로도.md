## CODE 1

import sys

input = sys.stdin.readline

a, b, c, m = map(int, input().split())

hour = fatigue = work = 0
while hour < 24:
    hour += 1
    if fatigue + a <= m:
        fatigue += a
        work += b
    else:
        fatigue -= c
        if fatigue < 0:
            fatigue = 0

print(work)
