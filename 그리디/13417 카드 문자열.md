import sys
from collections import deque

input = sys.stdin.readline

for test_case in range(int(input())):
    n = int(input())
    a = list(input().split())

    que = deque(a.pop(0))
    for ch in a:
        if ch <= que[0]:
            que.appendleft(ch)
        else:
            que.append(ch)

    ans = ''.join(que)
    print(ans)
