## CODE 1

import sys

input = sys.stdin.readline

n = int(input())
a = list(map(int, input().split()))

res1 = res2 = 0
cnt = 0
for x in a:
    if x % 2 == 1:
        cnt += 1
    else:
        res1 += cnt
cnt = 0
for x in reversed(a):
    if x % 2 == 1:
        cnt += 1
    else:
        res2 += cnt

ans = min(res1, res2)
print(ans)
