## CODE 1

```python
import sys

input = sys.stdin.readline

s = input().strip()

cnt = 0
visited = [False] * len(s)
quack = 0
valid = True
for x in range(len(s)):
    quack = 0
    duck = False
    for y in range(x, len(s)):
        if not visited[y] and 'quack'[quack] == s[y]:
            visited[y] = True
            quack += 1
            if quack == 5:
                quack = 0
                duck = True
    if duck:
        cnt += 1
    if quack != 0:
        valid = False

ans = -1
if valid and all(visited):
    ans = cnt
print(ans)
```
