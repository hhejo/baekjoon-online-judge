## CODE 1

```python
import sys

input = sys.stdin.readline

fib = [0, 1]
for i in range(2, 46):
    fib.append(fib[i - 2] + fib[i - 1])

for test_case in range(int(input())):
    n = int(input())
    res = []
    for x in reversed(fib):
        if x <= n:
            res.append(x)
            n -= x
        if n == 0:
            break
    res.sort()
    print(*res)
```
