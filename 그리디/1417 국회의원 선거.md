## CODE 1

```python
import sys
from heapq import heapify, heappush, heappop

input = sys.stdin.readline

n = int(input())
dasom = -int(input())
heap = [-int(input()) for _ in range(n - 1)]
heapify(heap)
heappush(heap, 0)

ans = 0
while True:
    cnt = heappop(heap)
    if cnt >= dasom:
        if cnt == dasom:
            ans += 1
        break
    cnt += 1
    dasom -= 1
    ans += 1
    heappush(heap, cnt)

print(ans)
```
