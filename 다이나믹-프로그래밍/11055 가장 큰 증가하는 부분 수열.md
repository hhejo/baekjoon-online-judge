import sys

def solve():
    global N, A
    dp = list(A)

    # 0 ~ N-1
    for i in range(N):
        # 0 ~ i-1
        for j in range(i):
            if A[j] < A[i] and dp[j] + A[i] > dp[i]:
                dp[i] = dp[j] + A[i]  # ê°±ì‹ 

    ans = max(dp)  # ì´ ì¤‘ì—ì„œ ê°€ì¥ í° ê°’
