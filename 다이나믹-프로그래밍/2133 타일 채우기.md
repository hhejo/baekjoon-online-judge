import sys

input = sys.stdin.readline

N = int(input())


# í’€ì´
def solve():
    global N
    dp = [0] * 31
    dp[0] = 1
    dp[1] = 0
    dp[2] = 3
    dp[3] = 0

    for i in range(4, N + 1):
        dp[i] += 3 * dp[i - 2]
        for j in range(4, i + 1, 2):
            dp[i] += 2 * dp[i - j]

    ans = dp[N]
    print(ans)
    return


solve()
