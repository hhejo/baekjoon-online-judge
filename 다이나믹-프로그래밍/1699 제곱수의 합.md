## CODE 1

import sys

input = sys.stdin.readline

N = int(input())


# í’€ì´
def solve():
    global N
    dp = [i for i in range(N + 1)]

    for i in range(1, N + 1):
        for j in range(1, int(i ** 0.5) + 1):
            if dp[i] > dp[i - j * j] + 1:
                dp[i] = dp[i - j * j] + 1

    ans = dp[N]
    print(ans)
    return


solve()
