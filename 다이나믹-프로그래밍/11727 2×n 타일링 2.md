import sys

input = sys.stdin.readline

# ÏûÖÎ†•
N = int(input())


# ÌíÄÏù¥
def solve():
    global N
    MOD = 10007
    dp = [0] * (N + 1)

    def calc(n):
        if n == 0 or n == 1:
            return 1
        if dp[n] > 0:
            return dp[n]
        dp[n] = (calc(n - 2) * 2 + calc(n - 1)) % MOD
        return dp[n]

    ans = calc(N)
    print(ans)
    return


solve()
