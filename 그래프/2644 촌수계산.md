## CODE 1

```python
import sys
from collections import deque

input = sys.stdin.readline


def bfs(n, s, g, graph):
    que = deque([s])
    v = [-1] * (n + 1)
    v[s] = 0
    while que:
        cur = que.popleft()
        if cur == g:
            return v[cur]
        for nxt in graph[cur]:
            if v[nxt] == -1:
                que.append(nxt)
                v[nxt] = v[cur] + 1
    return -1


n = int(input())
s, g = map(int, input().split())
m = int(input())
graph = [[] for _ in range(n + 1)]
for _ in range(m):
    x, y = map(int, input().split())
    graph[x].append(y)
    graph[y].append(x)

ans = bfs(n, s, g, graph)
print(ans)

```
