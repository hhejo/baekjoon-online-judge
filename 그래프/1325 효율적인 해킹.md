        cur = stk.pop()

                cnt += 1

import sys

input = sys.stdin.readline


def dfs(start):
    global n, graph
    stk = [start]
    v = [False] * (n + 1)
    v[start] = True
    cnt = 1
    while stk:
        cur = stk.pop()
        for nxt in graph[cur]:
            if not v[nxt]:
                stk.append(nxt)
                v[nxt] = True
                cnt += 1
    return cnt


n, m = map(int, input().split())
graph = [[] for _ in range(n + 1)]
for _ in range(m):
    a, b = map(int, input().split())
    graph[b].append(a)

res = 0
ans = []
for i in range(1, n + 1):
    cnt = dfs(i)
    if res == cnt:
        ans.append(i)
    elif res < cnt:
        res = cnt
        ans.clear()
        ans.append(i)
print(*sorted(ans))
