### CODE

```python
import sys
from collections import deque


# 풀이
def solve():
    global N, arr

    # BFS
    def bfs(x):
        que = deque([x])
        visited = [False] * (N + 1)
        visited[x] = True
        cnt = 1  # 해킹한 수
        while que:
            cur = que.popleft()
            for nxt in arr[cur]:
                if not visited[nxt]:
                    que.append(nxt)
                    visited[nxt] = True
                    cnt += 1
        return cnt

    ans = []
    max_cnt = 1
    for sx in range(1, N + 1):
        cnt = bfs(sx)
        # 현재까지 최대 해킹 수보다 크면 arr 배열의 값 전부 지우고 컴퓨터 번호 삽입
        if max_cnt < cnt:
            max_cnt = cnt
            ans.clear()
            ans.append(sx)
        # 현재까지 최대 해킹 수와 같으면 그냥 컴퓨터 번호 삽입
        elif max_cnt == cnt:
            ans.append(sx)
    print(*sorted(ans))


# 입력
N, M = map(int, sys.stdin.readline().split(' '))  # 자연수 N, M
arr = [[] * (N + 1) for _ in range(N + 1)]  # 신뢰 관계 리스트
for _ in range(M):
    A, B = map(int, sys.stdin.readline().split(' '))
    arr[B].append(A)  # A가 B를 신뢰 (B에서 A로 갈 수 있음)

# 풀이
solve()

```

### 해설

A가 B를 신뢰한다는 말은 단방향 간선으로 보면 된다.

탐색을 진행하면서, 현재까지 가장 제일 많이 감염을 시켰다면, 나머지는 지운다.
