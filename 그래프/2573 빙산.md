## CODE 1

```python
import sys

input = sys.stdin.readline

N, M = map(int, input().split())  # 3 <= N, M <= 300
g = [list(map(int, input().split())) for _ in range(N)]  # 0 <= g[x][y] <= 10


# 풀이
def solve():
    global N, M, g
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)

    def get_sea_count(cx, cy):
        sea = 0
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if g[nx][ny] == 0:
                sea += 1
        return sea

    def melt():
        a = [[0] * M for _ in range(N)]
        for x in range(1, N - 1):
            for y in range(1, M - 1):
                a[x][y] = get_sea_count(x, y)
        for x in range(1, N - 1):
            for y in range(1, M - 1):
                g[x][y] -= a[x][y]
                if g[x][y] < 0:
                    g[x][y] = 0
        return

    def dfs(sx, sy, v):
        stk = [(sx, sy)]
        v[sx][sy] = True
        while stk:
            cx, cy = stk.pop()
            for i in range(4):
                nx, ny = cx + dx[i], cy + dy[i]
                if 0 < nx < N - 1 and 0 < ny < M - 1 and g[nx][ny] > 0 and not v[nx][ny]:
                    stk.append((nx, ny))
                    v[nx][ny] = True
        return

    def get_icebergs_count():
        cnt = 0
        v = [[False] * M for _ in range(N)]
        for x in range(1, N - 1):
            for y in range(1, M - 1):
                if g[x][y] > 0 and not v[x][y]:
                    dfs(x, y, v)
                    cnt += 1
        return cnt

    ans = 0
    year = 0
    while True:
        cnt = get_icebergs_count()
        if cnt >= 2:
            ans = year
            break
        elif cnt == 0:
            ans = 0
            break
        melt()
        year += 1

    print(ans)
    return


solve()

```

```python
import sys

input = sys.stdin.readline


def melt():
    global N, M, g
    a = [[0] * M for _ in range(N)]
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
    for x in range(1, N - 1):
        for y in range(1, M - 1):
            if g[x][y] == 0:
                continue
            cnt = 0
            for i in range(4):
                nx, ny = x + dx[i], y + dy[i]
                if g[nx][ny] == 0:
                    cnt += 1
            a[x][y] = cnt
    for x in range(1, N - 1):
        for y in range(1, M - 1):
            if g[x][y] == 0:
                continue
            g[x][y] -= a[x][y]
            if g[x][y] < 0:
                g[x][y] = 0
    return


def dfs(sx, sy, v):
    global N, M, g
    stk = [(sx, sy)]
    v[sx][sy] = True
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
    while stk:
        cx, cy = stk.pop()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 < nx < N - 1 and 0 < ny < M - 1 and g[nx][ny] != 0 and not v[nx][ny]:
                stk.append((nx, ny))
                v[nx][ny] = True
    return


def get_icebergs():
    global N, M, g
    v = [[False] * M for _ in range(N)]
    cnt = 0
    for x in range(N):
        for y in range(M):
            if g[x][y] != 0 and not v[x][y]:
                dfs(x, y, v)
                cnt += 1
    return cnt


N, M = map(int, input().split())
g = [list(map(int, input().split())) for _ in range(N)]

ans = 0
year = 0
while True:
    cnt = get_icebergs()
    if cnt >= 2:
        ans = year
        break
    elif cnt == 0:
        break
    melt()
    year += 1

print(ans)

```
