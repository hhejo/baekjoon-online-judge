import sys
from collections import deque

input = sys.stdin.readline

M, N, H = map(int, input().split())
g = []
for _ in range(H):
    g.append([list(map(int, input().split())) for _ in range(N)])


# 풀이
def solve():
    global M, N, H, g

    def get_starts(g: list) -> list:
        starts = []
        for x in range(H):
            for y in range(N):
                for z in range(M):
                    if g[x][y][z] == 1:
                        starts.append((x, y, z))
        return starts

    def bfs(starts: list) -> None:
        que = deque(starts)
        for sx, sy, sz in starts:
            v[sx][sy][sz] = 0
        dx = (-1, 0, 0, 0, 0, 1)
        dy = (0, -1, 0, 1, 0, 0)
        dz = (0, 0, 1, 0, -1, 0)
        while que:
            cx, cy, cz = que.popleft()
            for i in range(6):
                nx, ny, nz = cx + dx[i], cy + dy[i], cz + dz[i]
                if 0 <= nx < H and 0 <= ny < N and 0 <= nz < M and g[nx][ny][nz] == 0 and v[nx][ny][nz] == -1:
                    que.append((nx, ny, nz))
                    v[nx][ny][nz] = v[cx][cy][cz] + 1
                    g[nx][ny][nz] = 1
        return

    def get_max_days(g: list, v: list) -> int:
        max_days = 0
        for x in range(H):
            for y in range(N):
                for z in range(M):
                    if g[x][y][z] == 0:
                        return -1
                    if max_days < v[x][y][z]:
                        max_days = v[x][y][z]
        return max_days

    starts: list = get_starts(g)
    v = [[[-1] * M for _ in range(N)] for _ in range(H)]
    bfs(starts)
    ans: int = get_max_days(g, v)
    print(ans)
    return


solve()
