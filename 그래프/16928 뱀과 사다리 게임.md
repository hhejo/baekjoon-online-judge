```python
import sys
from collections import deque

input = sys.stdin.readline

N, M = map(int, input().split())
g = [0] * 101
for _ in range(N):
    X, Y = map(int, input().split())
    g[X] = Y
for _ in range(M):
    U, V = map(int, input().split())
    g[U] = V


# í’€ì´
def solve():
    global g

    def bfs(start):
        que = deque([start])
        v = [-1] * 101
        v[start] = 0
        while que:
            cur = que.popleft()
            if cur == 100:
                return v[100]
            for i in range(1, 7):
                nxt = cur + i
                if 1 <= nxt <= 100 and v[nxt] == -1:
                    if g[nxt] > 0:
                        nxt = g[nxt]
                    if v[nxt] == -1:
                        v[nxt] = v[cur] + 1
                        que.append(nxt)
        return

    ans = bfs(1)
    print(ans)
    return


solve()

```
