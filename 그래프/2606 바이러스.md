import sys
from collections import deque

input = sys.stdin.readline

N = int(input())
K = int(input())
g = [[] for _ in range(N + 1)]
for _ in range(K):
    X, Y = map(int, input().split())
    g[X].append(Y)
    g[Y].append(X)


# í’€ì´
def solve():
    global N, g

    def bfs(start):
        que = deque([start])
        v = [False] * (N + 1)
        v[start] = True
        cnt = 0
        while que:
            cur = que.popleft()
            for nxt in g[cur]:
                if not v[nxt]:
                    que.append(nxt)
                    v[nxt] = True
                    cnt += 1
        return cnt

    ans = bfs(1)
    print(ans)
    return


solve()
