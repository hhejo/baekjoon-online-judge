## CODE

```python
import sys
from collections import deque

input = sys.stdin.readline

N, M = map(int, input().split())
g = [list(input().rstrip()) for _ in range(N)]


def get_max_distance(v):
    res = 0
    for row in v:
        res = max(res, *row)
    return res


def bfs(sx, sy, v):
    global N, M, g
    que = deque([(sx, sy)])
    v[sx][sy] = 0
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
    while que:
        cx, cy = que.popleft()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < N and 0 <= ny < M and g[nx][ny] == 'L' and v[nx][ny] == -1:
                que.append((nx, ny))
                v[nx][ny] = v[cx][cy] + 1
    return


ans = 0
for x in range(N):
    for y in range(M):
        if g[x][y] == 'W':
            continue
        v = [[-1] * M for _ in range(N)]
        bfs(x, y, v)
        res = get_max_distance(v)
        ans = max(ans, res)

print(ans)

```
