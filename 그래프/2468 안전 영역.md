import sys

#############################
sys.stdin = open('input.txt')
#############################


# 풀이
def solve():
    global N, arr

    # DFS
    def dfs(x, y):
        stk = [(x, y)]  # 스택에 푸시
        visited[x][y] = True  # 방문 처리
        # 스택이 빌 때까지
        while stk:
            cx, cy = stk.pop()  # 현재 위치 팝
            # 다음 위치 상, 우, 하, 좌
            for nx, ny in ((cx - 1, cy), (cx, cy + 1), (cx + 1, cy), (cx, cy - 1)):
                # 유효 범위, 높이가 h 보다 높음, 방문하지 않음
                if 0 <= nx < N and 0 <= ny < N and arr[nx][ny] > h and not visited[nx][ny]:
                    stk.append((nx, ny))  # 스택에 푸시
                    visited[nx][ny] = True  # 방문 처리
        return

    max_cnt = 0  # 물에 잠기지 않는 안전한 영역의 최대 개수
    # 높이 0 이상 100 이하 반복
    for h in range(101):
        visited = [[False] * N for _ in range(N)]  # 방문 여부
        cnt = 0  # 안전 영역 개수
        # 모든 좌표를 시작 위치로 반복
        for sx in range(N):
            for sy in range(N):
                # 해당 시작 위치 높이가 h 보다 큼, 방문하지 않음
                if arr[sx][sy] > h and not visited[sx][sy]:
                    dfs(sx, sy)  # DFS
                    cnt += 1  # 안전 영역 증가
        max_cnt = max(max_cnt, cnt)  # 물에 잠긴 높이가 h일 때 현재까지 최대 안전 영역 개수
    print(max_cnt)  # 정답 출력


# 입력
N = int(sys.stdin.readline())  # 2차원 배열의 행과 열의 개수
arr = [list(map(int, sys.stdin.readline().split(' '))) for _ in range(N)]

# 풀이
solve()

import sys
from collections import deque

#############################
sys.stdin = open('input.txt')
#############################


# 풀이
def solve():
    global N, arr

    # BFS
    def bfs(x, y):
        que = deque([(x, y)])  # 큐에 인큐
        visited[x][y] = True  # 방문 처리
        # 큐가 빌 때까지
        while que:
            cx, cy = que.popleft()  # 현재 위치 디큐
            # 다음 위치 상, 우, 하, 좌
            for nx, ny in ((cx - 1, cy), (cx, cy + 1), (cx + 1, cy), (cx, cy - 1)):
                # 유효 범위, 높이가 h 보다 높음, 방문하지 않음
                if 0 <= nx < N and 0 <= ny < N and arr[nx][ny] > h and not visited[nx][ny]:
                    que.append((nx, ny))  # 큐에 인큐
                    visited[nx][ny] = True  # 방문 처리
        return

    max_cnt = 0  # 물에 잠기지 않는 안전한 영역의 최대 개수
    # 높이 0 이상 100 이하 반복
    for h in range(101):
        visited = [[False] * N for _ in range(N)]  # 방문 여부
        cnt = 0  # 안전 영역 개수
        # 모든 좌표를 시작 위치로 반복
        for sx in range(N):
            for sy in range(N):
                # 해당 시작 위치 높이가 h 보다 큼, 방문하지 않음
                if arr[sx][sy] > h and not visited[sx][sy]:
                    bfs(sx, sy)  # BFS
                    cnt += 1  # 안전 영역 증가
        max_cnt = max(max_cnt, cnt)  # 물에 잠긴 높이가 h일 때 현재까지 최대 안전 영역 개수
    print(max_cnt)  # 정답 출력


# 입력
N = int(sys.stdin.readline())  # 2차원 배열의 행과 열의 개수
arr = [list(map(int, sys.stdin.readline().split(' '))) for _ in range(N)]

# 풀이
solve()
