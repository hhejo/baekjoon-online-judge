```python
import sys

input = sys.stdin.readline
sys.setrecursionlimit(10**8)

#


def dfs(cur):
    global g, v, order
    for nxt in g[cur]:
        if v[nxt]:
            continue
        order += 1
        v[nxt] = order
        dfs(nxt)
    return


N, M, R = map(int, input().split())
g = [[] for _ in range(N + 1)]
for _ in range(M):
    U, V = map(int, input().split())
    g[U].append(V)
    g[V].append(U)

for i in range(1, N + 1):
    g[i].sort()

v = [0] * (N + 1)
v[R] = 1
order = 1
dfs(R)

print(*v[1:], sep='\n')

```
