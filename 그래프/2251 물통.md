import sys
from collections import deque

#############################
sys.stdin = open('input.txt')
#############################

input = sys.stdin.readline

arr = list(map(int, input().split()))


def bfs(start):
    que = deque([(0, 0, start[2])])
    v = [[[False] * 201 for _ in range(201)] for _ in range(201)]
    v[0][0][start[2]] = True

    while que:
        cx, cy, cz = que.popleft()
        if cx == 0:
            ans.append(cz)

        # a -> b
        ny = cy + cx
        nx = 0
        if ny > arr[1]:
            ny = arr[1]
            nx = (cy + cx) - arr[1]
        if not v[nx][ny][cz]:
            que.append((nx, ny, cz))
            v[nx][ny][cz] = True

        # a -> c
        nz = cz + cx
        nx = 0
        if nz > arr[2]:
            nz = arr[2]
            nx = (cz + cx) - arr[2]
        if not v[nx][cy][nz]:
            que.append((nx, cy, nz))
            v[nx][cy][nz] = True

        # b -> a
        nx = cx + cy
        ny = 0
        if nx > arr[0]:
            nx = arr[0]
            ny = (cx + cy) - arr[0]
        if not v[nx][ny][cz]:
            que.append((nx, ny, cz))
            v[nx][ny][cz] = True

        # b -> c
        nz = cz + cy
        ny = 0
        if nz > arr[2]:
            nz = arr[2]
            ny = (cz + cy) - arr[2]
        if not v[cx][ny][nz]:
            que.append((cx, ny, nz))
            v[cx][ny][nz] = True

        # c -> a
        nx = cx + cz
        nz = 0
        if nx > arr[0]:
            nx = arr[0]
            nz = (cx + cz) - arr[0]
        if not v[nx][cy][nz]:
            que.append((nx, cy, nz))
            v[nx][cy][nz] = True

        # c -> b
        ny = cy + cz
        nz = 0
        if ny > arr[1]:
            ny = arr[1]
            nz = (cy + cz) - arr[1]
        if not v[cx][ny][nz]:
            que.append((cx, ny, nz))
            v[cx][ny][nz] = True

    return


ans = []

bfs(arr)
ans.sort()

print(*ans)
