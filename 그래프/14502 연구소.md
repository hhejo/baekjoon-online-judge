## CODE

```python
import sys
from copy import deepcopy

input = sys.stdin.readline

N, M = map(int, input().split())
g = [list(map(int, input().split())) for _ in range(N)]


def get_safe_area(a):
    cnt = 0
    for row in a:
        cnt += row.count(0)
    return cnt


def virus_spread(v, starts):
    global N, M, g
    a = deepcopy(g)
    stk = [*starts]
    for sx, sy in starts:
        v[sx][sy] = True
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
    while stk:
        cx, cy = stk.pop()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < N and 0 <= ny < M and a[nx][ny] == 0 and not v[nx][ny]:
                stk.append((nx, ny))
                a[nx][ny] = 2
                v[nx][ny] = True
    return a


def build_wall(wall, start):
    global N, M, g, ans, starts
    if wall == 3:
        v = [[False] * M for _ in range(N)]
        a = virus_spread(v, starts)
        safe = get_safe_area(a)
        ans = max(ans, safe)
        return
    for z in range(start, N * M):
        r, c = divmod(z, M)
        if g[r][c] == 0:
            g[r][c] = 1
            build_wall(wall + 1, z)
            g[r][c] = 0
    return


def get_virus_starts():
    global N, M, g
    starts = []
    for x in range(N):
        for y in range(M):
            if g[x][y] == 2:
                starts.append((x, y))
    return starts


starts = get_virus_starts()
ans = 0
build_wall(0, 0)
print(ans)

```
