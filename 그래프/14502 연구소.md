import sys
from copy import deepcopy

input = sys.stdin.readline

N, M = map(int, input().split())
g = [list(map(int, input().split())) for _ in range(N)]


def get_safe_area(a):
    cnt = 0
    for row in a:
        cnt += row.count(0)
    return cnt


def virus_spread(v, starts):
    global N, M, g
    a = deepcopy(g)
    stk = [*starts]
    for sx, sy in starts:
        v[sx][sy] = True
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
    while stk:
        cx, cy = stk.pop()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < N and 0 <= ny < M and a[nx][ny] == 0 and not v[nx][ny]:
                stk.append((nx, ny))
                a[nx][ny] = 2
                v[nx][ny] = True
    return a


def build_wall(wall, start):
    global N, M, g, ans, starts
    if wall == 3:
        v = [[False] * M for _ in range(N)]
        a = virus_spread(v, starts)
        safe = get_safe_area(a)
        ans = max(ans, safe)
        return
    for z in range(start, N * M):
        r, c = divmod(z, M)
        if g[r][c] == 0:
            g[r][c] = 1
            build_wall(wall + 1, z)
            g[r][c] = 0
    return


def get_virus_starts():
    global N, M, g
    starts = []
    for x in range(N):
        for y in range(M):
            if g[x][y] == 2:
                starts.append((x, y))
    return starts


starts = get_virus_starts()
ans = 0
build_wall(0, 0)
print(ans)

```

import sys

input = sys.stdin.readline
sys.setrecursionlimit(10 ** 9)


def get_safe_area(virus_graph):
    global n, m
    cnt = 0
    for x in range(n):
        for y in range(m):
            if virus_graph[x][y] == 0:
                cnt += 1
    return cnt


def dfs(virus_graph):
    global n, m, starts
    dx, dy = (-1, 1, 0, 0), (0, 0, -1, 1)
    stk = list(starts)
    visited = [[False] * m for _ in range(n)]
    for sx, sy in starts:
        visited[sx][sy] = True
    while stk:
        cx, cy = stk.pop()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < n and 0 <= ny < m and virus_graph[nx][ny] == 0 and not visited[nx][ny]:
                stk.append((nx, ny))
                virus_graph[nx][ny] = 2
                visited[nx][ny] = True
    return


def set_virus():
    global n, m, graph
    virus_graph = []
    for row in graph:
        virus_graph.append(row[:])
    dfs(virus_graph)
    return virus_graph


def set_wall(start, wall):
    global n, m, graph, ans
    if wall == 3:
        virus_graph = set_virus()
        cnt = get_safe_area(virus_graph)
        ans = max(ans, cnt)
        return
    for i in range(start, n * m):
        x, y = divmod(i, m)
        if graph[x][y] == 0:
            graph[x][y] = 1
            set_wall(i + 1, wall + 1)
            graph[x][y] = 0
    return


n, m = map(int, input().split())
graph = [list(map(int, input().split())) for _ in range(n)]

starts = []
for x in range(n):
    for y in range(m):
        if graph[x][y] == 2:
            starts.append((x, y))

ans = 0

set_wall(0, 0)

print(ans)

```
