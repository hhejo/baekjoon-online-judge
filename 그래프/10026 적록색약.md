import sys

input = sys.stdin.readline
sys.setrecursionlimit(10 ** 6)

N = int(input())
g = [list(input().strip()) for _ in range(N)]


# 풀이
def solve():
    global N, g
    dx = (-1, 0, 1, 0)
    dy = (0, 1, 0, -1)

    def dfs(cx, cy, weak):
        for i in range(4):
            nx = cx + dx[i]
            ny = cy + dy[i]
            if 0 <= nx < N and 0 <= ny < N and not v[nx][ny]:
                if weak:
                    if (g[cx][cy] == g[nx][ny] == 'B') or (g[cx][cy] == 'R' and g[nx][ny] in 'RG') or (g[cx][cy] == 'G' and g[nx][ny] in 'RG'):
                        v[nx][ny] = True
                        dfs(nx, ny, weak)
                else:
                    if g[cx][cy] == g[nx][ny]:
                        v[nx][ny] = True
                        dfs(nx, ny, weak)
        return

    ans = []
    v = [[False] * N for _ in range(N)]
    cnt = 0
    for x in range(N):
        for y in range(N):
            if not v[x][y]:
                cnt += 1
                v[x][y] = True
                dfs(x, y, False)
    ans.append(cnt)

    v = [[False] * N for _ in range(N)]
    cnt = 0
    for x in range(N):
        for y in range(N):
            if not v[x][y]:
                cnt += 1
                v[x][y] = True
                dfs(x, y, True)
    ans.append(cnt)
    print(*ans)
    return


solve()
