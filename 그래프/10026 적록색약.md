input = sys.stdin.readline


def dfs(x, y, v, weak):
    global N, g
    stk = [(x, y)]
    v[x][y] = True
    color = g[x][y]
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
    while stk:
        cx, cy = stk.pop()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < N and 0 <= ny < N and not v[nx][ny]:
                if not weak:
                    if g[nx][ny] == color:
                        stk.append((nx, ny))
                        v[nx][ny] = True
                else:
                    if color == 'B':
                        if g[nx][ny] == color:
                            stk.append((nx, ny))
                            v[nx][ny] = True
                    else:
                        if g[nx][ny] != 'B':
                            stk.append((nx, ny))
                            v[nx][ny] = True
    return


N = int(input())
g = [list(input().rstrip()) for _ in range(N)]

v = [[False] * N for _ in range(N)]
cnt1 = 0
for x in range(N):
    for y in range(N):
        if not v[x][y]:
            dfs(x, y, v, False)
            cnt1 += 1

    for y in range(N):
        if not v[x][y]:
            dfs(x, y, v, True)
            cnt2 += 1

import sys

input = sys.stdin.readline


def dfs(sx, sy, graph, v):
    global n
    dx, dy = (-1, 1, 0, 0), (0, 0, -1, 1)
    stk = [(sx, sy)]
    v[sx][sy] = True
    while stk:
        cx, cy = stk.pop()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < n and 0 <= ny < n and graph[sx][sy] == graph[nx][ny] and not v[nx][ny]:
                stk.append((nx, ny))
                v[nx][ny] = True
    return


def get_graph2():
    global n, graph
    graph2 = []
    for x in range(n):
        row = []
        for y in range(n):
            if graph[x][y] == 'G':
                row.append('R')
            else:
                row.append(graph[x][y])
        graph2.append(row)
    return graph2


n = int(input())
graph = [list(input().rstrip()) for _ in range(n)]

graph2 = get_graph2()

ans1 = ans2 = 0
v1 = [[False] * n for _ in range(n)]
v2 = [[False] * n for _ in range(n)]
for x in range(n):
    for y in range(n):
        if not v1[x][y]:
            ans1 += 1
            dfs(x, y, graph, v1)
        if not v2[x][y]:
            ans2 += 1
            dfs(x, y, graph2, v2)

print(ans1, ans2, sep=' ')
