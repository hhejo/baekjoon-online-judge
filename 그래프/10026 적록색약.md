import sys

input = sys.stdin.readline


def dfs(x, y, v, weak):
    global N, g
    stk = [(x, y)]
    v[x][y] = True
    color = g[x][y]
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
    while stk:
        cx, cy = stk.pop()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < N and 0 <= ny < N and not v[nx][ny]:
                if not weak:
                    if g[nx][ny] == color:
                        stk.append((nx, ny))
                        v[nx][ny] = True
                else:
                    if color == 'B':
                        if g[nx][ny] == color:
                            stk.append((nx, ny))
                            v[nx][ny] = True
                    else:
                        if g[nx][ny] != 'B':
                            stk.append((nx, ny))
                            v[nx][ny] = True
    return


N = int(input())
g = [list(input().rstrip()) for _ in range(N)]

v = [[False] * N for _ in range(N)]
cnt1 = 0
for x in range(N):
    for y in range(N):
        if not v[x][y]:
            dfs(x, y, v, False)
            cnt1 += 1

v = [[False] * N for _ in range(N)]
cnt2 = 0
for x in range(N):
    for y in range(N):
        if not v[x][y]:
            dfs(x, y, v, True)
            cnt2 += 1

print(cnt1, cnt2)
