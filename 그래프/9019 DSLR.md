import sys
from collections import deque

input = sys.stdin.readline


def calc_D(num):
    return (2 * num) % 10000, 'D'


def calc_S(num):
    return (num - 1) % 10000, 'S'


def calc_L(num):
    return (num % 1000) * 10 + num // 1000, 'L'


def calc_R(num):
    return (num % 10) * 1000 + (num // 10), 'R'


def bfs(start, goal):
    que = deque([(start, '')])
    v = [False] * 10000
    v[start] = True
    while que:
        cur, min_cmd = que.popleft()
        if cur == goal:
            return min_cmd
        for nxt, cmd in (calc_D(cur), calc_S(cur), calc_L(cur), calc_R(cur)):
            if not v[nxt]:
                que.append((nxt, min_cmd + cmd))
                v[nxt] = True
    return


for test_case in range(int(input())):
    A, B = map(int, input().split())
    ans = bfs(A, B)
    print(ans)
