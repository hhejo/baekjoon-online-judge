## CODE

```python
import sys
from collections import deque

input = sys.stdin.readline


# 풀이
def solve():
    global A, B

    def get_R(n):
        return (n % 10) * 1000 + n // 10

    def get_L(n):
        return (n % 1000) * 10 + n // 1000

    def get_S(n):
        return (n - 1) % 10000

    def get_D(n):
        return (n * 2) % 10000

    def bfs(start):
        que = deque([(start, '')])
        v = [-1] * 10000
        v[start] = 0
        calc = [get_D, get_S, get_L, get_R]
        while que:
            cur, log = que.popleft()
            if cur == B:
                return log
            for i in range(4):
                nxt = calc[i](cur)
                if v[nxt] == -1:
                    que.append((nxt, log + 'DSLR'[i]))
                    v[nxt] = v[cur] + 1
        return

    ans = bfs(A)
    print(ans)
    return


for test_case in range(int(input())):
    A, B = map(int, input().split())
    solve()

```
