## CODE 1

```python
import sys
from collections import deque

input = sys.stdin.readline

N, K = map(int, input().split())


def bfs(start):
    global N, K
    que = deque([start])
    v = [-1] * 100001
    v[start] = 0
    min_time = 100001
    cnt = 0
    while que:
        cur = que.popleft()
        if cur == K:
            min_time = v[cur]
            cnt += 1
            continue
        for nxt in (cur - 1, cur + 1, cur * 2):
            if 0 <= nxt < 100001:
                if v[nxt] == -1 or v[nxt] == v[cur] + 1:
                    que.append(nxt)
                    v[nxt] = v[cur] + 1
    return min_time, cnt


min_time, cnt = bfs(N)
print(min_time, cnt, sep='\n')

```
