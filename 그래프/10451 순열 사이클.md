### CODE 1

```python
import sys

input = sys.stdin.readline


# 테스트케이스 입력
T = int(input())


# 풀이
def solve():
    global N, arr

    # DFS
    def dfs(s):
        stk = [s]  # 스택에 시작 정점 푸시
        visited[s] = True  # 방문 처리
        # 스택 반복
        while stk:
            cur = stk.pop()  # 현재 정점 팝
            nxt = arr[cur]  # 이동할 다음 정점
            # 방문하지 않았다면
            if not visited[nxt]:
                stk.append(nxt)  # 다음 정점 푸시
                visited[nxt] = True  # 방문 처리
        return

    cnt = 0  # 순열 사이클 개수
    visited = [False] * (N + 1)  # 방문 여부
    # 1번 정점부터 N번 정점까지 반복
    for i in range(1, N + 1):
        # 방문하지 않았으면
        if not visited[i]:
            dfs(i)  # i번 정점부터 DFS
            cnt += 1  # 순열 사이클 개수 증가

    print(cnt)  # 정답 출력
    return


# 테스트케이스 풀이
for test_case in range(T):
    N = int(input())  # 순열의 크기
    arr = [0] + list(map(int, input().split()))  # 순열 배열
    # 풀이
    solve()

```

### CODE 2

```python
import sys

input = sys.stdin.readline
sys.setrecursionlimit(2000)


# 테스트케이스 입력
T = int(input())


# 풀이
def solve():
    global N, arr

    # DFS
    def dfs(s):
        # 정점 s 방문 했으면 종료
        if visited[s]:
            return
        visited[s] = True  # 정점 s 방문 처리
        dfs(arr[s])  # 정점 s와 연결된 다음 정점 DFS

    cnt = 0  # 순열 사이클 개수
    visited = [False] * (N + 1)  # 방문 여부
    # 1번 정점부터 N번 정점까지 반복
    for i in range(1, N + 1):
        # 방문하지 않았으면
        if not visited[i]:
            dfs(i)  # i번 정점부터 DFS
            cnt += 1  # 순열 사이클 개수 증가

    print(cnt)  # 정답 출력
    return


# 테스트케이스 풀이
for test_case in range(T):
    N = int(input())  # 순열의 크기
    arr = [0] + list(map(int, input().split()))  # 순열 배열
    # 풀이
    solve()
```
