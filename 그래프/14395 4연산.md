```python
import sys
from collections import deque

input = sys.stdin.readline

#


def add(num):
    return num + num, '+'


def mul(num):
    return num * num, '*'


def div(num):
    return 1, '/'


def bfs(start, goal):
    que = deque([(start, '')])
    v = set()
    v.add(start)
    while que:
        cur, operators = que.popleft()
        if cur == goal:
            return operators
        for nxt, op in (mul(cur), add(cur), div(cur)):
            if nxt <= 10 ** 9 and nxt not in v:
                que.append((nxt, operators + op))
                v.add(nxt)
    return -1


S, T = map(int, input().split())

ans = 0
if S != T:
    ans = bfs(S, T)
print(ans)

```
