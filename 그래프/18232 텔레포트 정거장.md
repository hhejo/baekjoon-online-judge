```python
import sys
from collections import deque

input = sys.stdin.readline

#


def bfs():
    global N, S, E, MAP
    que = deque([S])
    v = [-1] * (N + 1)
    v[S] = 0
    while que:
        cur = que.popleft()
        if cur == E:
            return v[cur]
        for nxt in MAP[cur]:
            if v[nxt] == -1:
                que.append(nxt)
                v[nxt] = v[cur] + 1
        for nxt in (cur - 1, cur + 1):
            if 0 < nxt <= N and v[nxt] == -1:
                que.append(nxt)
                v[nxt] = v[cur] + 1
    return


N, M = map(int, input().split())
S, E = map(int, input().split())
MAP = [[] for _ in range(N + 1)]
for _ in range(M):
    X, Y = map(int, input().split())
    MAP[X].append(Y)
    MAP[Y].append(X)

ans = bfs()
print(ans)

```
