## CODE 1

```python
import sys
from collections import deque


# 풀이
def solve():
    global N, M, arr  # N, M, 미로

    # bfs
    def bfs():
        que = deque([(0, 0)])  # 큐
        # 큐가 빌 때까지
        while que:
            cx, cy = que.popleft()  # 현재 위치
            # 현재 위치가 도착 지점이면 현재 위치까지 지난 최소의 칸 수 리턴
            if cx == N - 1 and cy == M - 1:
                return arr[cx][cy]
            # 아니라면 현재 위치의 상 우 하 좌에 대해
            for nx, ny in ((cx - 1, cy), (cx, cy + 1), (cx + 1, cy), (cx, cy - 1)):
                # 범위 내에 있고 이동할 수 있는 칸이라면 큐에 넣고 이동할 위치 방문 처리
                # 현재 위치까지 이동한 칸 수 + 1
                if 0 <= nx < N and 0 <= ny < M and arr[nx][ny] == 1:
                    que.append((nx, ny))
                    arr[nx][ny] = arr[cx][cy] + 1
        return

    ans = bfs()  # (N, M)의 위치로 이동하는 최소의 칸 수
    print(ans)
    return


N, M = map(int, input().split())  # N, M
arr = [list(map(int, input())) for _ in range(N)]  # 미로

solve()  # 풀이

```

## 해설 1

최단거리를 큐를 이용해서 BFS로 구현한다. 방문 여부 배열은 굳이 사용하지 않아도 풀 수 있다.

import sys
from collections import deque

input = sys.stdin.readline

# 입력
N, M = map(int, input().split())
MAP = [list(map(int, input().strip())) for _ in range(N)]


# 풀이
def solve():
    global N, M, MAP

    # BFS
    def bfs(sx, sy):
        que = deque([(sx, sy)])
        visited[sx][sy] = 1
        while que:
            cx, cy = que.popleft()
            for nx, ny in ((cx - 1, cy), (cx, cy + 1), (cx + 1, cy), (cx, cy - 1)):
                if 0 <= nx < N and 0 <= ny < M and MAP[nx][ny] == 1 and visited[nx][ny] == 0:
                    que.append((nx, ny))
                    visited[nx][ny] = visited[cx][cy] + 1
        return visited[N - 1][M - 1]

    visited = [[0] * M for _ in range(N)]
    ans = bfs(0, 0)
    print(ans)
    return


# 풀이
solve()

```

import sys
from collections import deque

input = sys.stdin.readline

N, M = map(int, input().split())
g = [list(map(int, input().rstrip())) for _ in range(N)]


def bfs(sx, sy):
    global N, M, g
    que = deque([(sx, sy)])
    v = [[0] * M for _ in range(N)]
    v[sx][sy] = 1
    dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
    while que:
        cx, cy = que.popleft()
        if cx == N - 1 and cy == M - 1:
            return v[cx][cy]
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < N and 0 <= ny < M and g[nx][ny] == 1 and not v[nx][ny]:
                que.append((nx, ny))
                v[nx][ny] = v[cx][cy] + 1
    return


ans = bfs(0, 0)
print(ans)

```

def bfs(sx, sy, gx, gy):

    que = deque([(sx, sy)])

    v[sx][sy] = 1
    dx, dy = (-1, 1, 0, 0), (0, 0, -1, 1)
    while que:
        cx, cy = que.popleft()

            if 0 <= nx < N and 0 <= ny < M and MAP[nx][ny] == 1 and not v[nx][ny]:
                que.append((nx, ny))
                v[nx][ny] = v[cx][cy] + 1
    return v[gx][gy]


N, M = map(int, input().split())
