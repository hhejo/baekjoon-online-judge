input = sys.stdin.readline

N, M = map(int, input().split())
g = [[] for _ in range(N + 1)]
for _ in range(M):
    U, V = map(int, input().split())
    g[U].append(V)
    g[V].append(U)

def solve():
    global N, g

    def bfs(start):
        que = deque([start])
        v[start] = True
        while que:
            cur = que.popleft()
            for nxt in g[cur]:
                if not v[nxt]:
                    que.append(nxt)
                    v[nxt] = True
        return

    ans = 0
    v = [False] * (N + 1)
    for i in range(1, N + 1):
        if not v[i]:
            bfs(i)
            ans += 1
    print(ans)
    return
