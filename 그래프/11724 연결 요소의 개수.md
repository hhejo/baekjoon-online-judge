import sys

def solve():
    global N, arr

    # DFS
    def dfs(x):
        stk = [x]  # 스택에 시작점 푸시
        visited[x] = True  # 방문 처리
        # 스택이 빌 때까지
        while stk:
            cur = stk.pop()  # 현재 위치를 스택에서 팝
            # 현재 위치에서 연결된 모든 다음 정점에 대해
            for nxt in arr[cur]:
                # 방문하지 않았다면
                if not visited[nxt]:
                    stk.append(nxt)  # 다음 정점을 스택에 푸시
                    visited[nxt] = True  # 방문 처리
        return

    cnt = 0  # 연결 요소의 개수
    visited = [False] * (N + 1)  # 해당 정점 방문 여부
    # N번 반복
    for x in range(1, N + 1):
        # 방문하지 않았다면
        if not visited[x]:
            dfs(x)  # DFS
            cnt += 1  # 연결 요소 개수 증가
    # 정답 출력
    print(cnt)
    return


# 입력
N, M = map(int, sys.stdin.readline().split(' '))  # 정점의 개수, 간선의 개수
arr = [[] for _ in range(N + 1)]  # 연결 그래프
# 간선의 개수만큼 반복
for _ in range(M):
    U, V = map(int, sys.stdin.readline().split(' '))  # 간선의 양 끝점

    arr[V].append(U)  # 간선 연결

solve()

import sys
from collections import deque


# 풀이
def solve():
    global N, arr

    # BFS
    def bfs(x):
        que = deque([x])  # 큐에 시작점 푸시
        visited[x] = True  # 방문 처리
        # 큐가 빌 때까지
        while que:
            cur = que.popleft()  # 현재 위치를 큐에서 디큐
            # 현재 위치에서 연결된 모든 다음 정점에 대해
            for nxt in arr[cur]:
                # 방문하지 않았다면
                if not visited[nxt]:
                    que.append(nxt)  # 다음 정점을 큐에 인큐
                    visited[nxt] = True  # 방문 처리
        return

    cnt = 0  # 연결 요소의 개수
    visited = [False] * (N + 1)  # 해당 정점 방문 여부
    # N번 반복
    for x in range(1, N + 1):
        # 방문하지 않았다면
        if not visited[x]:
            bfs(x)  # BFS
            cnt += 1  # 연결 요소 개수 증가
    # 정답 출력
    print(cnt)
    return


# 입력
N, M = map(int, sys.stdin.readline().split(' '))  # 정점의 개수, 간선의 개수
arr = [[] for _ in range(N + 1)]  # 연결 그래프
# 간선의 개수만큼 반복
for _ in range(M):
    U, V = map(int, sys.stdin.readline().split(' '))  # 간선의 양 끝점
    arr[U].append(V)  # 간선 연결
    arr[V].append(U)  # 간선 연결

# 풀이
solve()

import sys
from collections import deque

input = sys.stdin.readline

N, M = map(int, input().split())
g = [[] for _ in range(N + 1)]
for _ in range(M):
    U, V = map(int, input().split())
    g[U].append(V)
    g[V].append(U)


# 풀이
def solve():
    global N, g

    def bfs(start):
        que = deque([start])
        v[start] = True
        while que:
            cur = que.popleft()
            for nxt in g[cur]:
                if not v[nxt]:
                    que.append(nxt)
                    v[nxt] = True
        return

    ans = 0
    v = [False] * (N + 1)
    for i in range(1, N + 1):
        if not v[i]:
