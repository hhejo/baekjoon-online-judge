## CODE

```python
import sys
from collections import deque

input = sys.stdin.readline


N, M, R = map(int, input().split())
g = [[] for _ in range(N + 1)]
v = [0] * (N + 1)
cnt = 2

for _ in range(M):
    A, B = map(int, input().split())
    g[A].append(B)
    g[B].append(A)

for i in range(N + 1):
    g[i].sort()


def bfs():
    global cnt, v
    que = deque([R])
    v[R] = 1
    while que:
        cur = que.popleft()
        for nxt in g[cur]:
            if not v[nxt]:
                que.append(nxt)
                v[nxt] = cnt
                cnt += 1
    return


bfs()

for n in v[1:]:
    print(n)

```
