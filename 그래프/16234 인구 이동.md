```python
import sys

input = sys.stdin.readline


def move_population(v, dic):
    global N, g
    for x in range(N):
        for y in range(N):
            if v[x][y]:
                g[x][y] = dic[v[x][y]]
    return


def dfs(sx, sy, v, union):
    global N, L, R, g, dx, dy
    stk = [(sx, sy)]
    v[sx][sy] = union
    cnt = 0
    acc = 0
    while stk:
        cx, cy = stk.pop()
        cnt += 1
        acc += g[cx][cy]
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < N and 0 <= ny < N and L <= abs(g[cx][cy] - g[nx][ny]) <= R and not v[nx][ny]:
                stk.append((nx, ny))
                v[nx][ny] = union
    return acc // cnt


def get_each_union_and_population(v):
    global N, L, R, g, union
    dic = {}
    for x in range(N):
        for y in range(N):
            for i in range(4):
                nx, ny = x + dx[i], y + dy[i]
                if 0 <= nx < N and 0 <= ny < N and L <= abs(g[x][y] - g[nx][ny]) <= R and not v[x][y]:
                    union += 1
                    avg = dfs(x, y, v, union)
                    dic[union] = avg
    return dic


N, L, R = map(int, input().split())
g = [list(map(int, input().split())) for _ in range(N)]

dx, dy = (-1, 0, 1, 0), (0, 1, 0, -1)
ans = 0

while True:
    union = 0
    v = [[0] * N for _ in range(N)]

    population_dic = get_each_union_and_population(v)

    if union == 0:
        break
    else:
        move_population(v, population_dic)
        ans += 1

print(ans)

```

```python
import sys

input = sys.stdin.readline

dx, dy = (-1, 1, 0, 0), (0, 0, -1, 1)


def set_union(n, graph, v, u):
    for x in range(n):
        for y in range(n):
            if v[x][y] != 0:
                graph[x][y] = u[v[x][y]]
    return


def dfs(n, l, r, graph, v, num, sx, sy):
    global dx, dy
    stk = [(sx, sy)]
    v[sx][sy] = num
    acc = graph[sx][sy]
    cnt = 1
    while stk:
        cx, cy = stk.pop()
        for i in range(4):
            nx, ny = cx + dx[i], cy + dy[i]
            if 0 <= nx < n and 0 <= ny < n and l <= abs(graph[nx][ny] - graph[cx][cy]) <= r and v[nx][ny] == 0:
                stk.append((nx, ny))
                v[nx][ny] = num
                acc += graph[nx][ny]
                cnt += 1
    return acc // cnt


def get_union(n, l, r, graph, v):
    global dx, dy
    u = {}
    num = 0
    for x in range(n):
        for y in range(n):
            for i in range(4):
                nx, ny = x + dx[i], y + dy[i]
                if 0 <= nx < n and 0 <= ny < n and l <= abs(graph[nx][ny] - graph[x][y]) <= r and v[nx][ny] == 0:
                    num += 1
                    avg = dfs(n, l, r, graph, v, num, x, y)
                    u[num] = avg
    return u


n, l, r = map(int, input().split())
graph = [list(map(int, input().split())) for _ in range(n)]

ans = 0
while True:
    v = [[0] * n for _ in range(n)]
    u = get_union(n, l, r, graph, v)
    if not u:
        break
    ans += 1
    set_union(n, graph, v, u)
print(ans)
```
