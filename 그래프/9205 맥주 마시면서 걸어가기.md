def bfs(sx, sy, gx, gy, arr):
    global N
    que = deque([(sx, sy)])
    v = [False] * (N + 1)
    while que:
        cx, cy = que.popleft()
        if abs(cx - gx) + abs(cy - gy) <= 1000:
            return 'happy'
        for i in range(N):
            if not v[i]:
                nx, ny = arr[i]
                if abs(nx - cx) + abs(ny - cy) <= 1000:
                    que.append((nx, ny))
                    v[i] = True
    return 'sad'


for test_case in range(int(input())):
