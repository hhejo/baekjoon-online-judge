# 10101 삼각형 외우기



```python
angle = []
for _ in range(3):
    angle.append(int(input()))
angle.sort()  # 일단 정렬

if angle[0] == angle[1] == angle[2] == 60:
    print('Equilateral')
elif sum(angle) == 180:
    if (angle[0] == angle[1] or angle[1] == angle[2]):    
        print('Isosceles')
    else:
        print('Scalene')
else:
    print('Error')
```



```python
angle = [int(input()) for _ in range(3)]
angle.sort()

if sum(angle) == 180:
    if angle[0] == angle[2]:
        print('Equilateral')
    elif (angle[0] == angle[1]) or (angle[1] == angle[2]):
        print('Isosceles')
    else:
        print('Scalene')
else:
    print('Error')
```
