## CODE 1

```python
import sys

input = sys.stdin.readline

n, m = map(int, input().split())
arr = list(map(int, input().split()))
arr.sort()

res = [0] * m


def combination(n, r, depth, start):
    global arr, res
    if depth == r:
        print(*res)
        return
    for i in range(start, n):
        res[depth] = arr[i]
        combination(n, r, depth + 1, i)
    return


combination(n, m, 0, 0)
```
