```python
import sys
import time

input = sys.stdin.readline

#


def sudoku(idx):
    global g, cr, cc, cs

    if idx >= 81:
        for r in g:
            print(*r, sep='')
        exit(0)

    row, col = idx // 9, idx % 9
    if g[row][col] != 0:
        sudoku(idx + 1)
        return

    for num in range(1, 10):
        if cr[row][num] or cc[col][num] or cs[row // 3][col // 3][num]:
            continue
        g[row][col] = num
        cr[row][num] = cc[col][num] = cs[row // 3][col // 3][num] = True
        sudoku(idx + 1)
        cs[row // 3][col // 3][num] = cc[col][num] = cr[row][num] = False
        g[row][col] = 0
    return


g = [list(map(int, input().rstrip())) for _ in range(9)]

cr = [[False] * 10 for _ in range(9)]  # í–‰
cc = [[False] * 10 for _ in range(9)]  # ì—´
cs = [[[False] * 10 for _ in range(3)] for _ in range(3)]  # 3x3 ì •ì‚¬ê°í˜•
for x in range(81):
    row, col = x // 9, x % 9
    cr[row][g[row][col]] = True
    cc[col][g[row][col]] = True
    cs[row // 3][col // 3][g[row][col]] = True

sudoku(0)

```
