import sys

input = sys.stdin.readline


def permutation(cnt, start):
    global a, used, p
    if cnt == 6:
        print(*p)
        return
    for i in range(start, k - 6 + cnt + 1):
        if not used[i]:
            used[i] = True
            p[cnt] = a[i]
            permutation(cnt + 1, i + 1)
            used[i] = False
