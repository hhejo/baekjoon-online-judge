## CODE

```python
import sys

input = sys.stdin.readline


def permutation(cnt, start):
    global a, used, p
    if cnt == 6:
        print(*p)
        return
    for i in range(start, k - 6 + cnt + 1):
        if not used[i]:
            used[i] = True
            p[cnt] = a[i]
            permutation(cnt + 1, i + 1)
            used[i] = False


# 테스트 케이스 입력
while True:
    tmp = list(map(int, input().split()))
    if tmp[0] == 0:
        break
    k = tmp[0]
    a = tmp[1:]

    p = [0] * 6
    used = [False] * 50
    permutation(0, 0)
    print()

```
