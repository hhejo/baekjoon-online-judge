import sys

input = sys.stdin.readline


def permutation(cnt, start):
    global a, used, p
    if cnt == 6:
        print(*p)
        return
    for i in range(start, k - 6 + cnt + 1):
        if not used[i]:
            used[i] = True
            p[cnt] = a[i]
            permutation(cnt + 1, i + 1)
            used[i] = False


# 테스트 케이스 입력
while True:
    tmp = list(map(int, input().split()))
    if tmp[0] == 0:
        break
    k = tmp[0]
    a = tmp[1:]

    p = [0] * 6
    used = [False] * 50
    permutation(0, 0)
    print()

```

import sys

input = sys.stdin.readline


def permutation(cnt, idx):
    global K, a, p, res, res_idx
    if cnt == 6:
        res[res_idx] = ' '.join(map(str, p))
        res_idx += 1
        return
    for i in range(idx, K):
        p[cnt] = a[i]
        permutation(cnt + 1, i + 1)
        p[cnt] = 0
    return


while True:
    arr = list(map(int, input().split()))
    if arr[0] == 0:
        break
    K = arr[0]
    a = arr[1:]

    p = [0] * 6
    res = [''] * 924
    res_idx = 0
    permutation(0, 0)
    print(*res[:res_idx], sep='\n')
    print()

```
