```python
import sys

input = sys.stdin.readline

n = int(input())
a = list(map(int, input().split()))

res = [0] * n
visited = [False] * n
maxx = 0


def dfs(depth):
    global n, a, res, visited, maxx
    if depth == n:
        acc = sum([abs(res[i] - res[i + 1]) for i in range(n - 1)])
        maxx = max(maxx, acc)
        return
    for i in range(n):
        if not visited[i]:
            res[depth] = a[i]
            visited[i] = True
            dfs(depth + 1)
            visited[i] = False
    return


dfs(0)
print(maxx)
```

```python
import sys
from itertools import permutations

input = sys.stdin.readline

n = int(input())
a = list(map(int, input().split()))

maxx = 0

for perm in permutations(a, n):
    acc = sum([abs(perm[i] - perm[i + 1]) for i in range(n - 1)])
    maxx = max(maxx, acc)

print(maxx)
```
