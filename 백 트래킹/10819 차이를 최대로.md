## CODE

```python
import sys

input = sys.stdin.readline

N = int(input())
a = list(map(int, input().split()))


# 식 계산값 리턴
def calculate():
    global arr
    res = sum([abs(arr[i] - arr[i + 1]) for i in range(N - 1)])
    return res


# 순열
def next_permutation(depth):
    global used, arr, idx
    if depth >= N:
        res[idx] = calculate()
        idx += 1
        return
    for i in range(N):
        if not used[i]:
            used[i] = True
            arr[depth] = a[i]
            next_permutation(depth + 1)
            used[i] = False
    return


a.sort()
res = [0] * 40321
idx = 0
arr = list(a)
used = [False] * N
next_permutation(0)
print(max(res))

```
