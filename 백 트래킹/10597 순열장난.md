```python
import sys

input = sys.stdin.readline

s = input().strip()

n = len(s)
if len(s) > 9:
    n = 9 + (len(s) - 9) // 2

res = []


def dfs(depth):
    global s, n, res
    if depth == len(s):
        print(' '.join(res))
        exit(0)
    if s[depth] != '0' and s[depth] not in res:
        res.append(s[depth])
        dfs(depth + 1)
        res.pop()
    if s[depth] != '0' and int(s[depth:depth+2]) <= n and s[depth:depth+2] not in res:
        res.append(s[depth:depth+2])
        dfs(depth + 2)
        res.pop()
    return


dfs(0)
```
