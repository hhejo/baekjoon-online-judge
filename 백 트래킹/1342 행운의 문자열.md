```python
import sys
from collections import Counter

input = sys.stdin.readline

s = list(input().strip())

cnt = 0
counter = Counter(s)


def dfs(depth, prev):
    global s, cnt, counter
    if depth == len(s):
        cnt += 1
        return
    for ch in counter.keys():
        if prev == ch or counter[ch] == 0:
            continue
        counter[ch] -= 1
        dfs(depth + 1, ch)
        counter[ch] += 1
    return


dfs(0, '')
print(cnt)
```
