## CODE 1

```python
import sys

input = sys.stdin.readline

n, m = map(int, input().split())

arr = list(range(1, n + 1))
visited = [False] * n
res = [0] * m


def combination(n, r, depth):
    global arr, visited, res
    if depth == r:
        print(*res)
        return
    for i in range(n):
        res[depth] = arr[i]
        combination(n, r, depth + 1)
    return


combination(n, m, 0)
```
